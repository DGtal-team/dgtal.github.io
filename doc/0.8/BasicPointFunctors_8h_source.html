426  Point2DEmbedderIn3D( const TDomain3D &aDomain3DImg,
427  const Point &anOriginPoint, const typename Space::RealPoint & anNormalVector,
428  const typename Point::Component &anWidth,
429  const Point &aDefautPoint = Point(0,0,0)): myDomain(aDomain3DImg),
430  myDefaultPoint (aDefautPoint)
431  {
432  double d = -anNormalVector[0]*anOriginPoint[0] - anNormalVector[1]*anOriginPoint[1] - anNormalVector[2]*anOriginPoint[2];
433  typename Space::RealPoint pRefOrigin;
434  if(anNormalVector[0]!=0){
435  pRefOrigin [0]= -d/anNormalVector[0];
436  pRefOrigin [1]= 0.0;
437  pRefOrigin [2]= 0.0;
438  if(pRefOrigin==anOriginPoint){
439  pRefOrigin[1]=-1.0;
440  }
441  }else if (anNormalVector[1]!=0){
442  pRefOrigin [0]= 0.0;
443  pRefOrigin [1]= -d/anNormalVector[1];
444  pRefOrigin [2]= 0.0;
445  if(pRefOrigin==anOriginPoint){
446  pRefOrigin[0]=-1.0;
447  }
448  }else if (anNormalVector[2]!=0){
449  pRefOrigin [0]= 0.0;
450  pRefOrigin [1]= 0.0;
451  pRefOrigin [2]= -d/anNormalVector[2];
452  if(pRefOrigin==anOriginPoint){
453  pRefOrigin[0]=-1.0;
454  }
455  }
456  typename Space::RealPoint uDir1;
457  uDir1=(pRefOrigin-anOriginPoint)/((pRefOrigin-anOriginPoint).norm());
458  typename Space::RealPoint uDir2;
459  uDir2[0] = uDir1[1]*anNormalVector[2]-uDir1[2]*anNormalVector[1];
460  uDir2[1] = uDir1[2]*anNormalVector[0]-uDir1[0]*anNormalVector[2];
461  uDir2[2] = uDir1[0]*anNormalVector[1]-uDir1[1]*anNormalVector[0];
462

464

610  BasicDomainSubSampler(const TDomain &aSourceDomain, const std::vector<TValue> &aGridSize,
611  const Point &aGridShift): mySourceDomain(aSourceDomain),

617  Dimension sizeInDim = (aSourceDomain.upperBound()[dim] -

623  domainUpperBound);

627  myGridSampleDomain = TDomain(Point::diagonal(0), upperGrid);

645  Point operator()(const Point& aPoint) const
646  {
647  Point ptRes = Point::diagonal(0);
648  if(!myNewDomain.isInside(aPoint)){
649  trace.error() << " The point is not in the source domain: "<< aPoint << std::endl;
650  return ptRes;
651  }

654  ptRes[dim] = static_cast<TInteger>(floor(NumberTraits<TInteger>::castToDouble(aPoint[dim])*
655  NumberTraits<TValue>::castToDouble(myGridSize[dim])));
656  }

660  // we are looking for a point inside the domain
