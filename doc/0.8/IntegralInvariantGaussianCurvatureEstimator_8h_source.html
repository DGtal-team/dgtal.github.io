169  void init( const double & h, const double & r )
170  {
171  double r3 = r * r * r;
172  double r6 = r3 * r3;
173  d6_PIr6 = 6.0 / ( M_PI * r6 );
174  d8_5r = 8.0 / ( 5.0 * r );
175  double h2 = h * h;
176  dh5 = h2 * h2 * h;
177  }

180  {
181  Matrix3x3 cp_matrix = aInput;
182  cp_matrix *= dh5;
183  Value k1,k2;
184  Matrix3x3 eigenVectors;
185  Vector3 eigenValues;
186  evalk1k2( cp_matrix, eigenVectors, eigenValues, k1, k2 );
187  return k1 * k2;
188  }

191  void evalk1k2(
192  Matrix3x3 & matrix,
193  Matrix3x3 & eigenVectors,
194  Vector3 & eigenValues,
195  Value & k1,
196  Value & k2 )
197  {
198  EigenValues3D< Value >::getEigenDecomposition( matrix, eigenVectors, eigenValues );
199 
200  ASSERT ( eigenValues[ 0 ] == eigenValues[ 0 ] ); // NaN
201  ASSERT ( (eigenValues[ 0 ] <= eigenValues[ 2 ]) && (eigenValues[ 0 ] <= eigenValues[ 1 ]) && (eigenValues[ 1 ] <= eigenValues[ 2 ]) );
