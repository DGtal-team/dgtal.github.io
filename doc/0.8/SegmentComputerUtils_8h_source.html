233 template <typename SC>
234 bool maximalSymmetricExtension(SC& s,
235  const typename SC::ConstIterator& begin,
236  const typename SC::ConstIterator& end,
237  IteratorType ) {
238 
239  bool flagOk = true;
240  bool flagForward = true;
241  //while the extension is possible
242  //at the front and (then) at the back
243  while (flagOk) {
244  if (flagForward) {
245  flagForward = false;
246  if ( s.end() != end ) flagOk = s.extendFront();
247  else flagOk = false;
248  } else {
249  flagForward = true;
250  if ( s.begin() != begin ) flagOk = s.extendBack();
251  else flagOk = false;
252  }
253  }
254  //extend one more time if s.begin() == begin
255  if (s.begin() != begin ) {
256  if (s.extendBack()) return !s.extendFront();
257  else return false;
258  } else {
259  return !flagForward;
260  }
261 
262 }

511 template <typename SC>
512 void mostCenteredMaximalSegment(SC& s,
513  const typename SC::ConstIterator& i,
514  const typename SC::ConstIterator& begin,
515  const typename SC::ConstIterator& end,
516  DGtal::ForwardSegmentComputer )
517 {
518 
519  typedef typename SC::ConstIterator ConstIterator;
520  typedef typename SC::Reverse ReverseSegmentComputer;
521  typedef typename ReverseSegmentComputer::ConstIterator ConstReverseIterator;
522 
523  //get the first maximal segment passing through i
524 
525  firstMaximalSegment( s, i, begin, end, DGtal::ForwardSegmentComputer() );
526 
527  //get the next maximal segment while i is not at the middle of
528  //the current maximal segment.
529 
530  ConstIterator k( s.begin() );
531  while ( k != i ) {
532 
533  if ( isNotEmpty<ConstIterator>(s.end(),end) ) {
534 
535  //backward extension
536  ConstIterator it( s.end() ); ++it;
537  ConstReverseIterator rit( it );
538  ConstReverseIterator rend( s.begin() );
539  ReverseSegmentComputer r( s.getReverse() );
540  longestSegment(r, rit, rend);
541  ConstIterator newBegin = r.end().base();
542  ASSERT( newBegin != s.begin() );
543 
544  while ( ( k != getMiddleIterator(newBegin, s.end() ) )
545  &&( k != i ) ) {
546  ++k;
547  }
548  if ( k != i ) {
549 
550  //get the next maximal segment
551  longestSegment(s, newBegin, end);
552 
553  }
554 
555  } else {
556  k = i;
557  }
558  }
559 }
