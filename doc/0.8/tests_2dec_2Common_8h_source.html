18  for (typename Container::Index jj=0; jj<container.cols(); jj++)

24 template <typename Container, typename Value>
25 bool
26 is_identity(const Container& container, const Value& value)
27 {
28  for (typename Container::Index ii=0; ii<container.rows(); ii++)
29  for (typename Container::Index jj=0; jj<container.cols(); jj++)
30  if ((ii != jj && container(ii,jj) != 0) || (ii == jj && container(ii,jj) != value))
31  return false;
32  return true;
33 }
