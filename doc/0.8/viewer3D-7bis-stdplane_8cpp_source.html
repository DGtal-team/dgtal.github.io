73 /*
74  Returns all the points in the \a domain that belongs to the standard
75  plane mu <= a*x+b*y+c*z < mu+|a|+|b|+|c|.
76 */
77 template <typename Domain>
78 std::vector<typename Domain::Point> pointsInStandardPlane
79 ( const Domain & domain,
80  typename Domain::Integer a,
81  typename Domain::Integer b,
82  typename Domain::Integer c,
83  typename Domain::Integer mu )
84 {
85  typedef typename Domain::Integer Integer;
86  typedef typename Domain::Point Point;
87  typedef typename Domain::ConstIterator ConstIterator;
88  std::vector<Point> pts;
89  Integer mup = mu + abs(a) + abs(b) + abs(c);
90  for ( ConstIterator it = domain.begin(), itE = domain.end();
91  it != itE; ++it )
92  {
93  Point p = *it;
94  Integer r = a * p[ 0 ] + b * p[ 1 ] + c * p[ 2 ];
95  if ( ( mu <= r ) && ( r < mup ) )
96  pts.push_back( p );
97  }
98  return pts;
99 }
