57 estimatorOnShapeDigitization( const string& name,
58  Shape & aShape,
59  const RealPoint& low, const RealPoint& up,
60  double h )
61 {
62  using namespace Z2i;
63 
64  trace.beginBlock ( ( "Curvature estimation on digitization of "
65  + name ). c_str() );
66 
67  // Creates a digitizer on the window (low, up).
68  GaussDigitizer<Space,Shape> dig;
69  dig.attach( aShape ); // attaches the shape.
70  dig.init( low, up, h );
71 
72  // The domain size is given by the digitizer
73  // according to the window and the step.
74  Domain domain = dig.getDomain();
75 
76  // Create cellular space
77  KSpace K;
78  bool ok = K.init( dig.getLowerBound(), dig.getUpperBound(), true );
79  if ( ! ok )
80  {
81  std::cerr << "[estimatorOnShapeDigitization]"
82  << " error in creating KSpace." << std::endl;
83  }
84  else
85  try {
86  // Extracts shape boundary
87  SurfelAdjacency<KSpace::dimension> SAdj( true );
88  SCell bel = Surfaces<KSpace>::findABel( K, dig, 10000 );
89  // Getting the consecutive surfels of the 2D boundary
90  std::vector<Point> points;
91  Surfaces<KSpace>::track2DBoundaryPoints( points, K, SAdj, dig, bel );
92  // Create GridCurve
93  GridCurve<KSpace> gridcurve( K );
94  gridcurve.initFromVector( points );
95  // Create range of incident points
96  typedef GridCurve<KSpace>::IncidentPointsRange Range;
97  typedef Range::ConstIterator ClassicIterator;
98  typedef Range::ConstCirculator CircularIterator;
99  Range r = gridcurve.getIncidentPointsRange();//building range
100  // Estimation
101  std::vector<double> estimations;
102  if (gridcurve.isOpen())
103  {
104  typedef StabbingCircleComputer<ClassicIterator> SegmentComputer;
105  typedef CurvatureFromDCAEstimator<SegmentComputer> SCEstimator;
106  typedef MostCenteredMaximalSegmentEstimator<SegmentComputer,SCEstimator> CurvatureEstimator;
107  SegmentComputer sc;
108  SCEstimator sce;
109  CurvatureEstimator estimator(sc, sce);
110  std::cout << "# open grid curve" << endl;
111  estimator.init( h, r.begin(), r.end() );
112  estimator.eval( r.begin(), r.end(), std::back_inserter(estimations) );
113  }
114  else
115  {
116  typedef StabbingCircleComputer<CircularIterator> SegmentComputer;
117  typedef CurvatureFromDCAEstimator<SegmentComputer> SCEstimator;
118  typedef MostCenteredMaximalSegmentEstimator<SegmentComputer,SCEstimator> CurvatureEstimator;
119  SegmentComputer sc;
120  SCEstimator sce;
121  CurvatureEstimator estimator(sc, sce);
122  std::cout << "# closed grid curve" << endl;
123  estimator.init( h, r.c(), r.c() );
124  estimator.eval( r.c(), r.c(), std::back_inserter(estimations) );
125  }
126  // Print (standard output)
127  std::cout << "# idx kappa" << endl;
128  unsigned int i = 0;
129  for ( ClassicIterator it = r.begin(), ite = r.end();
130  it != ite; ++it, ++i )
131  {
132  std::cout << i << " " << estimations.at(i) << std::endl;
133  }
134  }
135  catch ( InputException e )
136  {
137  std::cerr << "[estimatorOnShapeDigitization]"
138  << " error in finding a bel." << std::endl;
139  ok = false;
140  }
141  trace.emphase() << ( ok ? "Passed." : "Error." ) << endl;
142  trace.endBlock();
143  return ok;
144 }
