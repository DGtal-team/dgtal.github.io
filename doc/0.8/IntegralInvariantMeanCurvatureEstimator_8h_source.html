69 class MeanCurvatureFunctor3 : std::unary_function <double,double>
70 {
71 public:
72 
73  MeanCurvatureFunctor3(){}
74 
75  void init( const double & h, const double & r )
76  {
77  d8_3r = 8.0 / ( 3.0 * r );
78  double r2 = r * r;
79  d_4_PIr4 = 4.0 / ( M_PI * r2 * r2 );
80  dh3 = h * h * h;
81  }
82 
83  Quantity operator()(const Quantity & aInput) const
84  {
85  Quantity cp_quantity = aInput;
86  cp_quantity *= dh3;
87 
88  return d8_3r - d_4_PIr4 * cp_quantity;
89  }

105  void init( const double & h, const double & r )
106  {
107  d1_r2 = 1.0 / ( r * r );
108  dPI_2 = M_PI / 2.0;

114  {

116  cp_quantity *= dh2;
117 
118  return d3_r * ( dPI_2 - d1_r2 * cp_quantity );
119  }
120
