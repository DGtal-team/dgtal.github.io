00061 bool 
00062 DGtal::DigitalSurface2DSlice<TDigitalSurfaceTracker>::
00063 init( DigitalSurfaceTracker* tracker, Dimension i )
00064 {
00065   const KSpace & K = tracker->surface().space();
00066   Surfel start_s = tracker->current(); // the start surfel
00067   Surfel current_s = start_s; // the current surfel during the extraction
00068   Surfel next_s;     // the surfel after the current surfel during the extraction
00069   uint8_t code = 0;// the code turn when going to the next surfel.
00070   Dimension j = i; // the current tracking direction.
00071   if ( K.sOrthDir( start_s ) == i ) return false;
00072   do
00073     {
00074       mySurfels.push_back( current_s );
00075       bool direct = K.sDirect( current_s, j );
00076       code = tracker->adjacent( next_s, j, direct );
00077       if ( code == 0 ) break; // slice is open
00078       if ( code != 2 ) // 1 or 3, swap tracking dir and orthogonal dir.
00079         j = K.sOrthDir( current_s );
00080       tracker->move( next_s );
00081       current_s = next_s;
00082     }
00083   while ( next_s != start_s );
00084   // Test if we have looped or if the slice is open.
00085   myIsClosed = code != 0;
00086   unsigned int shift = 0;
00087   if ( ! myIsClosed )
00088     { // open slice 
00089       tracker->move( start_s );
00090       j = i;
00091       current_s = start_s;
00092       bool indirect = ! K.sDirect( current_s, j );
00093       while ( ( code = tracker->adjacent( next_s, j, indirect ) ) )
00094         { // valid movement;
00095           mySurfels.push_front( next_s );
00096           ++shift;
00097           if ( code != 2 ) // 1 or 3, swap tracking dir and orthogonal dir.
00098             j = K.sOrthDir( current_s );
00099           tracker->move( next_s );
00100           current_s = next_s;
00101           indirect = ! K.sDirect( current_s, j );
00102         }
00103     }
00104   myStart = begin() + shift;
00105   return true;
00106 }
