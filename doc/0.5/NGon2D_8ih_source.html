00134   // seek the vertices between the point, then compute the vector from one vertex to the next one.

00137   unsigned int intervale_upper = intervale_lower == ( myK -1 ) ? 0 : intervale_lower+1;
00138   double dist = myRadius*cos ( M_PI / myK );
00139   RealPoint2D s1 ( myRadius*cos(myPhi + intervale_lower*2.0*M_PI/myK),

00141   RealPoint2D s2 ( myRadius*cos(myPhi + intervale_upper*2.0*M_PI/myK),
00142        myRadius*sin(myPhi + intervale_upper*2.0*M_PI/myK) );
00143   RealPoint2D s3( s2[0] - s1[0], s2[1] - s1[1]);
00144   
00145   double line_angle = atan2f( (float)s3[ 1 ], (float)s3[ 0 ]);
00146   
00147   double rho = dist/(cos (t - line_angle - 0.5*M_PI));
