00378 {
00379 
00380   if ( myNextBefore != myPatternEnd ) {
00381     //Normal case
00382     //cout << "Case 1" << endl;
00383     ++myNextBefore;
00384 
00385   } else if ( isTrivial() ) {
00386     // In this case, it can be shorten only if there is more than one
00387     // repetition.
00388     //cout << "Case 2" << endl;
00389     if ( myNbRepeat == 1 ) return false;
00390     myPatternBegin = myPatternEnd = myNextBefore = ++myNextAfter;
00391     --myNbRepeat;
00392 
00393   } else if ( myNbRepeat >= 2 ) { 
00394     // Got more than one repetition, it suffices to consider the next
00395     // repetition of the main pattern with one less repetition.
00396     //cout << "Case 3" << endl;
00397     Size myOldMainPatternLength = mainPatternLength();
00398     myPatternBegin += myOldMainPatternLength;
00399     myPatternEnd   += myOldMainPatternLength;
00400     myNextAfter    += myOldMainPatternLength;
00401     myNextBefore = myPatternBegin;
00402     --myNbRepeat;
00403 
00404   } else {
00405     //Only one repetition, the slope is modified.
00406     Size myOldMainPatternLength = mainPatternLength();
00407     Size myOldLeftPatternLength = myLeftPatternLength;
00408     Size myOldRightPatternLength = myOldMainPatternLength - myOldLeftPatternLength;
00409 
00410     if ( prefixLength() >= myOldRightPatternLength ) {
00411       // A second Lower Leaning Point has been read in the prefix at
00412       // the end of the main pattern. The slope is simply reversed.
00413       //cout << "Case 4" << endl;
00414       myLeftPatternLength = myOldRightPatternLength;
00415       myPatternBegin     += myOldRightPatternLength;
00416       myPatternEnd       += myOldRightPatternLength;
00417       myNextBefore = myPatternEnd - myOldRightPatternLength + 1;
00418 
00419     } else if ( myOldLeftPatternLength < myOldRightPatternLength ) {
00420       // Remove one repetition of the left Berstel pattern.
00421       //cout << "Case 5" << endl;
00422       myPatternBegin += myOldLeftPatternLength;
00423       myNextBefore -= ( myOldLeftPatternLength - 1 );
00424       myNextAfter += myOldLeftPatternLength;
00425 
00426     } else if ( myOldLeftPatternLength > myOldRightPatternLength ) {
00427       // The complexity of the slope is modified.
00428       //cout << "Case 6" << endl;
00429       Size myNbBerstelRight = (myOldRightPatternLength > 1) ?
00430         myOldMainPatternLength / myOldRightPatternLength :
00431         myOldMainPatternLength - 1;
00432       Size myBerstelLeftLength = myOldMainPatternLength - 
00433         ( myNbBerstelRight * myOldRightPatternLength ); 
00434       // Right subpattern becomes the main pattern
00435       myNbRepeat = myNbBerstelRight;
00436       myPatternBegin += myBerstelLeftLength;
00437       myPatternEnd = myPatternBegin + myOldRightPatternLength - 1;
00438       myNextBefore = myPatternEnd - myBerstelLeftLength + 1;
00439       myNextAfter += myBerstelLeftLength;
00440       myLeftPatternLength = (myPatternBegin == myPatternEnd) ? 
00441         0 : myBerstelLeftLength;

00445       // DSS remains.

00973 template < typename TConstIterator, typename TInteger >
00974 inline
00975 void DGtal::CombinatorialDSS<TConstIterator,TInteger>::getArithmeticalDescription(
00976     Integer &a, Integer &b, Integer &mu, Integer &omega ) const
00977 {
00978   ConstPointIterator itBegin = pointBegin();
00979   while ( itBegin.getIndex() != myPatternBegin ) 
00980     itBegin++;
00981   ConstPointIterator itEnd = pointEnd();
00982   while ( itEnd.getIndex() != myPatternEnd+1 ) 
00983     itEnd--;
00984   ConstPointIterator it;
00985   Size myRightPatternLenght = mainPatternLength() - myLeftPatternLength;
00986   it = itBegin;
00987   for (int i=0; i<myRightPatternLenght; i++)
00988     it++;
00989   Point pb = *itBegin;
00990   Point pe = *itEnd;
00991   Point po = *it;
00992   Vector v = pe - pb;
00993   a = v.at(1);
00994   b = v.at(0);
00995   Integer r1 = a*pb.at(0) - b*pb.at(1);
00996   Integer r2 = a*po.at(0) - b*po.at(1);
00997   mu = min (r1, r2);
00998   omega = ((a>0) ? a : -a) + ((b>0) ? b : -b );
00999 }

01004 void DGtal::CombinatorialDSS<TConstIterator,TInteger>::computeLeaningPoints( 
01005               Point & uf, Point & ul,

01007 {

01010     ++it;

01025     {
01026       swap ( uf, lf );
01027       swap ( ul, ll );
