00256 DGtal::GradientColorMap<PValue,PDefaultPreset,PDefaultFirstColor,PDefaultLastColor>::getColor

00259   const Value & max,
00260   const Value & value )
00261 {
00262   if ( colors.size() < 2 )
00263     return Color::None;
00264   double scale = static_cast<double>( value - min ) / (max - min);
00265   const int intervals = (const int)colors.size() - 1;
00266   int upper_index = static_cast<int>( ceil( intervals * scale ) );
00267   if ( !upper_index ) // Special case when value == min.
00268     upper_index = 1;
00269   const Color & firstColor = colors[ upper_index - 1 ];
00270   const Color & lastColor = colors[ upper_index ];
00271   scale = ( scale * intervals ) - (upper_index - 1);
00272 
00273   const unsigned char red = static_cast<unsigned char>( firstColor.red() +
00274               scale * ( lastColor.red() - 
00275                   firstColor.red() ));
00276   const unsigned char green = static_cast<unsigned char>( firstColor.green() +
00277                 scale * ( lastColor.green() - 
00278                     firstColor.green() ));
00279   const unsigned char blue = static_cast<unsigned char>( firstColor.blue() +
00280                scale * ( lastColor.blue() - 
00281                    firstColor.blue() ));
00282   return Color( red, green, blue );
