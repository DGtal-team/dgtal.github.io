00119 DGtal::ImageLinearCellEmbedder<TKSpace, TImage, TEmbedder>::
00120 operator()( const Cell & cell ) const
00121 {
00122   ASSERT( this->isValid() );
00123   // embed first the spel related to the cell.
00124   Point p1( myPtrK->uCoords( cell ) );
00125   RealPoint x1( embed( p1 ) );
00126   ImageValue y1 = (*myPtrImage)( p1 );
00127   for ( typename KSpace::DirIterator qit = myPtrK->uOrthDirs( cell );
00128         qit != 0; ++qit )
00129     { // cell is closed along this dimension.
00130       // estimate coordinate by interpolation, looking for f(...)=0.
00131       Dimension k = *qit;
00132       Point p2( p1 ); --p2[ k ];
00133       RealPoint x2( embed( p2 ) );
00134       ImageValue y2 = (*myPtrImage)( p2 );
00135       x1[ k ] -= NumberTraits<ImageValue>::castToDouble( y1 - myIsoValue )
00136         * ( x2[ k ] - x1[ k ] )
00137         / NumberTraits<ImageValue>::castToDouble( y2 - y1 );
00138     }
00139   return x1;
