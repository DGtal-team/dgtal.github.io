00050 template <typename Quotient>
00051 bool
00052 equalCFrac( const std::vector<Quotient> & c1, const std::vector<Quotient> & c2 )
00053 {
00054   unsigned int s = c1.size() < c2.size() ? c1.size() : c2.size();
00055   if ( ( s != c1.size() ) && ( c1.back() != NumberTraits<Quotient>::ONE ) )
00056     return false;
00057   if ( ( s != c2.size() ) && ( c2.back() != NumberTraits<Quotient>::ONE ) )
00058     return false;
00059   for ( unsigned int i = 0; i < s; ++i )
00060     {
00061       Quotient q1 = c1[ i ];
00062       if ( ( s != c1.size() ) && ( i == s - 1 ) ) q1 += c1.back();
00063       Quotient q2 = c2[ i ];
00064       if ( ( s != c2.size() ) && ( i == s - 1 ) ) q2 += c2.back();
00065       if ( q1 != q2 ) return false;
00066     }
00067   return true;
00068 }
