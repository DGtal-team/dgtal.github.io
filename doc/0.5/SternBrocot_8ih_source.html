00649 inline
00650 typename DGtal::SternBrocot<TInteger, TQuotient>::Fraction
00651 DGtal::SternBrocot<TInteger, TQuotient>::fraction
00652 ( Integer p, Integer q,
00653   Fraction ancestor )
00654 {
00655   IntegerComputer<Integer> ic;
00656   Integer g = ic.gcd( p, q );
00657   if ( g != NumberTraits<Integer>::ZERO )
00658     {
00659       p /= g;
00660       q /= g;
00661     }
00662   // special case 1/0
00663   if ( ( p == NumberTraits<Integer>::ONE ) 
00664        && ( q == NumberTraits<Integer>::ZERO ) ) return oneOverZero();
00665   // other positive fractions
00666   while ( ! ancestor.equals( p, q ) )
00667     {
00668       ASSERT( ( p + q ) >= ( ancestor.p() + ancestor.q() )
00669               && "[ImaGene::SternBrocot::fraction] bad ancestor." );
00670       ancestor = ancestor.lessThan( p, q ) 
00671         ? ancestor.right()
00672         : ancestor.left();
00673     }
00674   return ancestor;
00675 }
