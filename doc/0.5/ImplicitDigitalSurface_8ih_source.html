00190 bool
00191 DGtal::ImplicitDigitalSurface<TKSpace,TPointPredicate>::isInside
00192 ( const Surfel & s ) const
00193 {
00194   Dimension k = myKSpace.sOrthDir( s );
00195   // checks if the surfel is on the space boundary.
00196   if ( myKSpace.sIsMax( s, k ) || myKSpace.sIsMin( s, k ) )
00197     return false;
00198   // p1 must be in the set and p2 must not be in the set.
00199   SCell spel1 = myKSpace.sDirectIncident( s, k );
00200   Point p1 = myKSpace.sCoords( spel1 );
00201   if ( myPointPredicate( p1 ) )
00202     {
00203       SCell spel2 = myKSpace.sIndirectIncident( s, k );
00204       Point p2 = myKSpace.sCoords( spel2 );
00205       return ! myPointPredicate( p2 );
00206     }
00207   return false;
00208 }
