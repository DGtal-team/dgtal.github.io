00935 typename DGtal::ArithmeticalDSS<TIterator,TInteger,connectivity>::PointD
00936 DGtal::ArithmeticalDSS<TIterator,TInteger,connectivity>::project( const Point & m ) const
00937 {
00938   //casts
00939   double a = (double) NumberTraits<TInteger>::castToInt64_t(myA);
00940   double b = (double) NumberTraits<TInteger>::castToInt64_t(myB);
00941   double mu = (double) NumberTraits<TInteger>::castToInt64_t(myMu);
00942   double omega = (double) NumberTraits<TInteger>::castToInt64_t(myOmega);
00943   double xm = (double) NumberTraits<TInteger>::castToInt64_t(m[0]);
00944   double ym = (double) NumberTraits<TInteger>::castToInt64_t(m[1]);
00945 
00946   double alpha = ( mu + (omega - 1.0)/2.0 );
00947   double d2 = ( a * a + b * b );
00948   double s = b * xm + a * ym;
00949   double xp = ( b * s + a * alpha ) / d2;
00950   double yp = ( a * s - b * alpha ) / d2;
00951   return PointD( xp, yp );
00952 }
