00053 template <typename Quotient>
00054 bool
00055 equalCFrac( const std::vector<Quotient> & c1, const std::vector<Quotient> & c2 )
00056 {
00057   unsigned int s = c1.size() < c2.size() ? c1.size() : c2.size();
00058   if ( ( s != c1.size() ) && ( c1.back() != NumberTraits<Quotient>::ONE ) )
00059     return false;
00060   if ( ( s != c2.size() ) && ( c2.back() != NumberTraits<Quotient>::ONE ) )
00061     return false;
00062   for ( unsigned int i = 0; i < s; ++i )
00063     {
00064       Quotient q1 = c1[ i ];
00065       if ( ( s != c1.size() ) && ( i == s - 1 ) ) q1 += c1.back();
00066       Quotient q2 = c2[ i ];
00067       if ( ( s != c2.size() ) && ( i == s - 1 ) ) q2 += c2.back();
00068       if ( q1 != q2 ) return false;
00069     }
00070   return true;
00071 }
