00118             {
00119               if ( x1[ j ] != x2[ j ] )

00121                   alreadyOnSameAxis = false;

00125                   if ( val_v2 != val_v1 )

00127                       d = i;
00128                     }
00129                   else
00130                     { // v1 is updated.
00131                       x1 = x2;
00132                       x2[ j ] = c;
00133                       d = j;
00134                     }
00135                 } // if ( x1[ j ] != x2[ j ] )
00136             } // for ( Dimension j = i + 1; j < K.dimension; ++j )
00137           if ( alreadyOnSameAxis )

00141 
00142   // (3) Check result.
00143   for ( Dimension i = 0; i < K.dimension; ++i )

00145       if ( ( i == d ) && ( x1[ i ] == x2[ i ] ) )
