00186 template <typename TKSpace, typename TDigitalSet>
00187 inline
00188 bool
00189 DGtal::DigitalSetBoundary<TKSpace,TDigitalSet>::isInside
00190 ( const Surfel & s ) const
00191 {
00192   Dimension k = myKSpace.sOrthDir( s );
00193   // checks if the surfel is on the space boundary.
00194   if ( myKSpace.sIsMax( s, k ) || myKSpace.sIsMin( s, k ) )
00195     return false;
00196   // p1 must be in the set and p2 must not be in the set.
00197   SCell spel1 = myKSpace.sDirectIncident( s, k ); 
00198   Point p1 = myKSpace.sCoords( spel1 );
00199   if ( myDigitalSet.find( p1 ) != myDigitalSet.end() )
00200     {
00201       SCell spel2 = myKSpace.sIndirectIncident( s, k ); 
00202       Point p2 = myKSpace.sCoords( spel2 );
00203       return ! ( myDigitalSet.find( p2 ) != myDigitalSet.end() );
00204     }
00205   return false;
00206 }
