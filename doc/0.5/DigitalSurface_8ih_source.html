00450   unsigned int nb = 0;

00462   while ( myUmbrellaComputer.surfel() != start );
00463   if ( code == 0 ) // open face
00464     { // Going back to count the number of incident vertices.

00471           code = myUmbrellaComputer.next();
00472         }
00473       while ( code != 0 );
00474       return Face( myUmbrellaComputer.state(), nb, false );
00475     }
00476   else             // closed face
