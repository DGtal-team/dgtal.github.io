00104 embedCell( const Cell & cell ) const
00105 {
00106   ASSERT( this->isValid() );
00107   // embed first the spel related to the cell.
00108   Point p1( myPtrK->uCoords( cell ) );
00109   RealPoint x1( embed( p1 ) );
00110   Value y1 = (*myPtrFct)( x1 );
00111   for ( typename KSpace::DirIterator qit = myPtrK->uOrthDirs( cell ); 
00112         qit != 0; ++qit )
00113     { // cell is closed along this dimension.
00114       // estimate coordinate by interpolation, looking for f(...)=0.
00115       Dimension k = *qit;
00116       Point p2( p1 ); --p2[ k ];
00117       RealPoint x2( embed( p2 ) );
00118       Value y2 = (*myPtrFct)( x2 );
00119       x1[ k ] -= y1 * ( x2[ k ] - x1[ k ] ) / ( y2 - y1 );
