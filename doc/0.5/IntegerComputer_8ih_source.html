00410 extendedEuclid( IntegerParamType a, IntegerParamType b, 
00411                 IntegerParamType c ) const
00412 {
00413   if( isZero( a ) )  return Point2I( NumberTraits<Integer>::ZERO, b * c );
00414   if( isZero( b ) )  return Point2I( a * c, NumberTraits<Integer>::ZERO );
00415 
00416   for ( unsigned int i = 0; i < 4; ++i )
00417     _m_bezout[ i ].clear();
00418 
00419   _m_bezout[ 0 ].push_back( abs( a ) );
00420   _m_bezout[ 0 ].push_back( abs( b ) );
00421   _m_bezout[ 1 ].push_back( NumberTraits<Integer>::ZERO );
00422   _m_bezout[ 2 ].push_back( NumberTraits<Integer>::ONE );
00423   _m_bezout[ 2 ].push_back( NumberTraits<Integer>::ZERO );
00424   _m_bezout[ 3 ].push_back( NumberTraits<Integer>::ZERO );
00425   _m_bezout[ 3 ].push_back( NumberTraits<Integer>::ONE );
00426   
00427   unsigned int k = 0; // index of the iteration during the computation.
00428   while( isNotZero( _m_bezout[ 0 ][ k+1 ] ) )
00429     {
00430       _m_bezout[ 1 ].push_back( _m_bezout[ 0 ][ k ] 
00431                                 / _m_bezout[ 0 ][ k+1 ] );
00432       _m_bezout[ 0 ].push_back( _m_bezout[ 0 ][ k ] 
00433                                 % _m_bezout[ 0 ][ k+1 ] );
00434       _m_bezout[ 2 ].push_back( _m_bezout[ 2 ][ k ] 
00435                                 - _m_bezout[ 1 ][ k+1 ]
00436                                 * _m_bezout[ 2 ][ k+1 ] );
00437       _m_bezout[ 3 ].push_back( _m_bezout[ 3 ][ k ] 
00438                                 - _m_bezout[ 1 ][ k+1 ]
00439                                 *_m_bezout[ 3 ][ k+1 ] );
00440       ++k;
00441     }
00442   
00443   _m_v[ 0 ] = _m_bezout[ 2 ][ k ];
00444   _m_v[ 1 ] = _m_bezout[ 3 ][ k ];
00445   
00446   if ( isNegative( a ) ) _m_v[ 0 ] = -_m_v[ 0 ];
00447   if ( isNegative( b ) ) _m_v[ 1 ] = -_m_v[ 1 ];
