00064 child( Quotient v )
00065 {
00066   typedef typename MapQuotientToNode::iterator Iterator;
00067   ASSERT( v != NumberTraits<Quotient>::ZERO );
00068   if ( v == NumberTraits<Quotient>::ONE ) 
00069     return ( this == instance().myOneOverZero )
00070       ? instance().myOneOverOne
00071       : this;
00072   Iterator itkey = myChildren.find( v );
00073   if ( itkey != myChildren.end() ) 
00074     return itkey->second;
00075   if ( this == instance().myOneOverZero )
00076     {
00077       Node* newNode = 
00078         new Node( (int) NumberTraits<Quotient>::castToInt64_t( v ),  // p' = v
00079                   NumberTraits<Integer>::ONE,              // q' = 1
00080                   v,                                       // u' = v
00081                   NumberTraits<Quotient>::ZERO,                // k' = 0
00082                   this );
00083       myChildren[ v ] = newNode;
00084       ++( instance().nbFractions );
00085       return newNode;
00086     }
00087   long int _v = NumberTraits<Quotient>::castToInt64_t( v );
00088   long int _u = NumberTraits<Quotient>::castToInt64_t( this->u );
00089   Integer _pp = origin() == instance().myOneOverZero 
00090     ? NumberTraits<Integer>::ONE
00091     : origin()->p;
00092   Integer _qq = origin() == instance().myOneOverZero
00093     ? NumberTraits<Integer>::ONE
00094     : origin()->q;
00095   Node* newNode = // p' = v*p - (v-1)*(p-p2)/(u-1)
00096     new Node( p * _v - ( _v - 1 ) * ( p - _pp ) / (_u - 1), 
00097               q * _v - ( _v - 1 ) * ( q - _qq ) / (_u - 1), 
00098               v,                           // u' = v
00099               k + NumberTraits<Quotient>::ONE, // k' = k+1
00100               this );
00101   myChildren[ v ] = newNode;
00102   ++( instance().nbFractions );
00103   return newNode;
00104 }
