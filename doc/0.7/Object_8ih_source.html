397 {
398  typedef std::vector<Vertex> Container;

400  typedef typename DigitalSet::ConstIterator DigitalSetConstIterator;
401  typedef typename DigitalSet::ConstIterator DigitalSetConstIterator;
402  //typedef typename Domain::Predicate Predicate;
403 
404  // Intermediate container that is fast writable.
405  const DigitalSet & mySet = pointSet();
406  Object<DigitalTopology, DigitalSet> output( topology(),
407  mySet.domainPointer() );
408  DigitalSet & outputSet = output.pointSet();
409 
410  // Loop on all points of the set.
411  Container tmp_local_points;
412  const DigitalSetConstIterator it_end = mySet.end();
413  for ( DigitalSetConstIterator it = mySet.begin();
414  it != it_end;
415  ++it )
416  {
417  std::back_insert_iterator< Container > back_ins_it( tmp_local_points );
418  // Computing neighborhood within domain.
419  topology().lambda().writeNeighbors
420  ( back_ins_it, *it, domain().predicate() );
421  // Checks if any point is not in the object.
422  const ContainerConstIterator itc_end( tmp_local_points.end() );
423  for ( ContainerConstIterator itc = tmp_local_points.begin();
424  itc != itc_end;
425  ++itc )
426  if ( pointSet().find( *itc ) == it_end )
427  {
428  outputSet.insertNew( *it );
429  break;
430  }
431  tmp_local_points.clear();
432  }
433  return output;
