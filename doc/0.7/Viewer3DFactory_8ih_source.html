338 template <typename Space, typename KSpace>
339 template <typename SpaceDom>
340 inline
341 void
342 DGtal::Viewer3DFactory<Space,KSpace>::draw( Viewer3D<Space, KSpace> & viewer,
343  const DGtal::HyperRectDomain<SpaceDom> & aDomain )
344 {
345  std::string mode = viewer.getMode( aDomain.className() );
346  ASSERT((SpaceDom::dimension==3)|| (SpaceDom::dimension==2));
347 
348  ASSERT((SpaceDom::dimension!=3) || (mode=="" || mode=="Grid" || mode=="Paving"|| mode=="PavingPoints"|| mode=="PavingGrids" ||
349  mode=="BoundingBox")||
350  ("DGtal::Display3DFactory<Space,KSpace>::draw( Display3D<Space, KSpace> & viewer, const DGtal::HyperRectDomain<Space> & aDomain ): Unknown mode "+mode)=="");
351 
352 
353  ASSERT((SpaceDom::dimension!=2) || (mode=="" || mode=="BoundingBox" || mode=="InterGrid"|| mode=="Grid") ||
354  ("DGtal::Display3DFactory<Space,KSpace>::draw( Display3D<Space, KSpace> & viewer, const DGtal::HyperRectDomain<Space> & aDomain ): Unknown mode "+mode)=="");
355 
356  if(SpaceDom::dimension == 3){
357  if ( mode == "BoundingBox" )
358  {
359  viewer.createNewLineList(aDomain.className());
360  drawAsBoundingBox( viewer, aDomain );
361  }else if( ( mode == "" ) || (mode == "Grid"))
362  {
363  viewer.createNewLineList(aDomain.className());
364  drawAsGrid( viewer, aDomain );
365  }
366  else if ( mode == "Paving" )
367  {
368  viewer.createNewCubeList( aDomain.className());
369  drawAsPaving( viewer, aDomain );
370  }else if ( mode == "PavingGrids" )
371  {
372  viewer.createNewLineList(aDomain.className());
373  viewer.createNewCubeList( aDomain.className());
374  drawAsGrid( viewer, aDomain );
375  drawAsPaving( viewer, aDomain );
376  }
377  }else if(SpaceDom::dimension == 2)
378  {
379  if (mode=="")
380  mode="BoundingBox";
381  viewer.addImage2DDomainD3D(aDomain, mode);
382  }
383 
384 }
