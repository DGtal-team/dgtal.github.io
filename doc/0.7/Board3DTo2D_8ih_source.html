183 template < typename Space, typename KSpace>
184 void
185 DGtal::Board3DTo2D<Space, KSpace>::LookAtMt(double mat[16],
186 double eyex, double eyey, double eyez,
187 double dirx, double diry, double dirz,
188 double upx, double upy, double upz)
189 {
190  double up[3]; up[0]= upx; up[1]= upy; up[2]= upz;
191 
192  double z[3]; z[0]= -dirx; z[1]= -diry; z[2]= -dirz; Display3D<Space,KSpace>::normalize(z);
193  double x[3]; Display3D<Space,KSpace>::cross (x, up, z); Display3D<Space,KSpace>::normalize(x);
194  double y[3]; Display3D<Space,KSpace>::cross (y, z, x); Display3D<Space,KSpace>::normalize(y);
195 
196  double m[16];
197  m[0] = x[0]; m[1] = x[1]; m[2] = x[2]; m[3] = 0;
198  m[4] = y[0]; m[5] = y[1]; m[6] = y[2]; m[7] = 0;
199  m[8] = z[0]; m[9] = z[1]; m[10] = z[2]; m[11] = 0;
200  m[12] = 0; m[13] = 0; m[14] = 0; m[15] = 1;
201 
202  double e[4]; e[0]= -eyex; e[1]= -eyey; e[2]= -eyez; e[3]= 1;
203  double eyePrime[4]; MulMt(eyePrime, m, e);
204 
205  TransposeMt(mat, m);
206  mat[12] = eyePrime[0]; mat[13] = eyePrime[1]; mat[14] = eyePrime[2];
207 }

231  0.00, 0.00, (ZNear + ZFar) / (ZNear - ZFar), -1.00,
232  0.00, 0.00, 2.0 * ZNear * ZFar / (ZNear - ZFar), 0.00 };

238  camera_upVector[0], camera_upVector[1], camera_upVector[2]);

241  {

245  for (unsigned short k=0; k<4; ++k)
246  sum += Projection[l+4*k]*Modelview[k+4*m];
247  matrix[l+4*m] = sum;

260 template < typename Space, typename KSpace>
261 void DGtal::Board3DTo2D<Space, KSpace>::project(double x3d, double y3d, double z3d, double &x2d, double &y2d)
262 {
263  double v[4], vs[4];
264  v[0]=x3d; v[1]=y3d; v[2]=z3d; v[3]=1.0;
265 
266  vs[0]=matrix[0 ]*v[0] + matrix[4 ]*v[1] + matrix[8 ]*v[2] + matrix[12 ]*v[3];
267  vs[1]=matrix[1 ]*v[0] + matrix[5 ]*v[1] + matrix[9 ]*v[2] + matrix[13 ]*v[3];
268  vs[2]=matrix[2 ]*v[0] + matrix[6 ]*v[1] + matrix[10]*v[2] + matrix[14 ]*v[3];
269  vs[3]=matrix[3 ]*v[0] + matrix[7 ]*v[1] + matrix[11]*v[2] + matrix[15 ]*v[3];
270 
271  vs[0] /= vs[3];
272  vs[1] /= vs[3];
273  vs[2] /= vs[3];
274 
275  vs[0] = vs[0] * 0.5 + 0.5;
276  vs[1] = vs[1] * 0.5 + 0.5;
277  vs[2] = vs[2] * 0.5 + 0.5;
278 
279  vs[0] = vs[0] * Viewport[2] + Viewport[0];
280  vs[1] = vs[1] * Viewport[3] + Viewport[1];
281 
282  x2d = vs[0];
283  y2d = Viewport[3]-vs[1];
284 }
