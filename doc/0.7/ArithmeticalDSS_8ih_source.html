481 template <typename TCoordinate, typename TInteger, unsigned short adjacency>
482 inline
483 unsigned short int
484 DGtal::ArithmeticalDSS<TCoordinate, TInteger, adjacency>::
485 isExtendableFront( const Point& aNewPoint ) const
486 {
487  Vector step = aNewPoint - myL;
488  Coordinate deviation = DGtal::ArithmeticalDSLKernel<TCoordinate,adjacency>::norm(step[0], step[1]);
489 
490  //if the two last points are confounded OK
491  if (deviation == NumberTraits<Coordinate>::ZERO)
492  return 9;
493 
494  //if the two last points are not connected KO
495  else if (deviation > NumberTraits<Coordinate>::ONE)
496  return 0;
497 
498  //if the first step does not exist yet OK
499  else if ( (myDSL.mySteps.first[0] == NumberTraits<Coordinate>::ZERO)
500  &&(myDSL.mySteps.first[1] == NumberTraits<Coordinate>::ZERO) )
501  return 1;
502 
503  //if the first step exists and
504  //if the second step does not exist
505  else if ( (myDSL.mySteps.second[0] == NumberTraits<Coordinate>::ZERO)
506  &&(myDSL.mySteps.second[1] == NumberTraits<Coordinate>::ZERO) )
507  {
508  //if the first step exists and is repeated OK
509  if (step == myDSL.mySteps.first)
510  return 2;
511  else
512  {
513  //if the two steps are compatible OK
514  Vector v = step - myDSL.mySteps.first;
515  if ( DGtal::ArithmeticalDSLKernel<TCoordinate,DGtal::ArithmeticalDSLKernel<TCoordinate,adjacency>::BackgroundAdjacency>
516  ::norm(v[0], v[1]) == NumberTraits<Coordinate>::ONE )
517  {
518  Integer r = remainder(aNewPoint);
519  //if weakly exterior on the left
520  if (r == myDSL.myLowerBound-NumberTraits<Integer>::ONE)
521  return 3;
522  //if weakly exterior on the right
523  else
524  {
525  ASSERT(r == myDSL.myUpperBound+NumberTraits<Integer>::ONE);
526  return 4;
527  }
528  }
529  else
530  return 0;
531  }
532  }
533  //if the two steps are initialized
534  else
535  {
536  //if there are only two steps
537  if ( (step == myDSL.mySteps.first) || (step == myDSL.mySteps.second ) )
538  {
539  Integer r = remainder(aNewPoint);
540 
541  //if strongly exterior KO
542  if ( (r < myDSL.myLowerBound-NumberTraits<Integer>::ONE)
543  ||(r > myDSL.myUpperBound+NumberTraits<Integer>::ONE) )
544  return 0;
545  //otherwise OK
546  else
547  {
548  if (r == myDSL.myLowerBound)
549  return 5;
550  else if (r == myDSL.myUpperBound)
551  return 6;
552  else if (r == myDSL.myLowerBound-NumberTraits<Integer>::ONE)
553  return 7;
554  else if (r == myDSL.myUpperBound+NumberTraits<Integer>::ONE)
555  return 8;
556  else
557  return 9;
558  }
559  }
560  //if there are more than two steps KO
561  else
562  return 0;
563  }
564 }

576 //-----------------------------------------------------------------------------
577 template <typename TCoordinate, typename TInteger, unsigned short adjacency>
578 inline
579 bool
580 DGtal::ArithmeticalDSS<TCoordinate, TInteger, adjacency>::
581 extendFront( const Point& aNewPoint )
582 {
583  //true if the DSS can be extended to aNewPoint
584  //false otherwise
585  bool flag = true;
586 
587  //code that tells how to update the DSS
588  unsigned short int res = isExtendableFront(aNewPoint);
589 
590  switch (res)
591  {
592  case 1: //first step init
593  myDSL.mySteps.first[0] = (aNewPoint[0] - myL[0]);
594  myDSL.mySteps.first[1] = (aNewPoint[1] - myL[1]);
595  myDSL.myA = myDSL.mySteps.first[1];
596  myDSL.myB = myDSL.mySteps.first[0];
597  myDSL.myLowerBound = remainder(myUf); //myUf doesn't change
598  myDSL.myUpperBound = remainder(myLf); //myLf doesn't change
599  myL = aNewPoint;
600  myUl = aNewPoint;
601  myLl = aNewPoint;
602  myDSL.myShift = DGtal::ArithmeticalDSLKernel<TCoordinate,adjacency>::shift(myDSL.myA, myDSL.myB);
603  break;
604  case 2: //first step repeated
605  myL = aNewPoint;
606  myUl = aNewPoint;
607  myLl = aNewPoint;
608  break;
609  case 3: //second step init on the left
610  myDSL.myA = ((myUl[1] - myUf[1]) + (aNewPoint[1] - myL[1]));
611  myDSL.myB = ((myUl[0] - myUf[0]) + (aNewPoint[0] - myL[0]));
612  myDSL.myLowerBound = remainder(myUf); //myUf doesn't change
613  myDSL.myUpperBound = remainder(myLl); //myLl doesn't change
614  myL = aNewPoint;
615  myUl = aNewPoint;
616  myLf = myLl;
617  myDSL.mySteps = DGtal::ArithmeticalDSLKernel<TCoordinate,adjacency>::steps(myDSL.myA, myDSL.myB);
618  myDSL.myShift = myDSL.mySteps.first-myDSL.mySteps.second;
619  break;
620  case 4: //second step init on the right
621  myDSL.myA = ((myLl[1] - myLf[1]) + (aNewPoint[1] - myL[1]));
622  myDSL.myB = ((myLl[0] - myLf[0]) + (aNewPoint[0] - myL[0]));
623  myDSL.myLowerBound = remainder(myUl); //myUl doesn't change
624  myDSL.myUpperBound = remainder(myLf); //myLf doesn't change
625  myL = aNewPoint;
626  myLl = aNewPoint;
627  myUf = myUl;
628  myDSL.mySteps = DGtal::ArithmeticalDSLKernel<TCoordinate,adjacency>::steps(myDSL.myA, myDSL.myB);
629  myDSL.myShift = myDSL.mySteps.first-myDSL.mySteps.second;
630  break;
631  case 5: //weakly interior on the left
632  myL = aNewPoint;
633  myUl = aNewPoint;
634  break;
635  case 6: //weakly interior on the right
636  myL = aNewPoint;
637  myLl = aNewPoint;
638  break;
639  case 7: //weakly exterior on the left
640  myDSL.myA = (aNewPoint[1] - myUf[1]);
641  myDSL.myB = (aNewPoint[0] - myUf[0]);
642  myDSL.myLowerBound = remainder(myUf); //myUf doesn't change
643  myDSL.myUpperBound = remainder(myLl); //myLl doesn't change
644  myL = aNewPoint;
645  myUl = aNewPoint;
646  myLf = myLl;
647  break;
648  case 8: //weakly exterior on the right
649  myDSL.myA = (aNewPoint[1] - myLf[1]);
650  myDSL.myB = (aNewPoint[0] - myLf[0]);

652  myDSL.myUpperBound = remainder(myLf); //myLf doesn't change

654  myLl = aNewPoint;

657  case 9: //strongly interior

659  break;

662  }

702 template <typename TCoordinate, typename TInteger, unsigned short adjacency>
703 inline
704 bool
705 DGtal::ArithmeticalDSS<TCoordinate, TInteger, adjacency>::
706 retractBack()
707 {
708  if (myF == myL)
709  return false;
710  else
711  {
712 
713  //next point computation
714  Point next = *++begin();
715  //if the next point is the last one
716  if (next == myL)
717  {
718  *this = ArithmeticalDSS(next);
719  return true;
720  }
721  //otherwise
722  else
723  {
724  //points used to update the DSS
725  Point bezoutPoint;
726 
727  //leaning points / parameters update
728  if (myF == myUf)
729  {
730  bezoutPoint = myUf + myDSL.myShift;
731  if ( retractUpdateLeaningPoints( Vector(myDSL.myB, myDSL.myA),
732  next, myL,
733  bezoutPoint,
734  myLf, myLl,
735  myUf, myUl ) )
736  retractUpdateParameters( myLf - bezoutPoint );
737 
738  }
739  if (myF == myLf)
740  {
741  bezoutPoint = myLf - myDSL.myShift;
742  if ( retractUpdateLeaningPoints( Vector(myDSL.myB, myDSL.myA),
743  next, myL,
744  bezoutPoint,
745  myUf, myUl,
746  myLf, myLl ) )
747  retractUpdateParameters( myUf - bezoutPoint );
748  }
749 
750  //first point update
751  myF = next;
752 
753  return true;
754  }
755  }
756 }

759 template <typename TCoordinate, typename TInteger, unsigned short adjacency>
760 inline
761 bool
762 DGtal::ArithmeticalDSS<TCoordinate, TInteger, adjacency>::
763 retractUpdateLeaningPoints( const Vector& aDirection,
764  const Point& aFirst,
765  const Point& aLast,
766  const Point& aBezout,
767  const Point& aFirstAtOppositeSide,
768  Point& aLastAtOppositeSide,
769  Point& aFirstAtRemovalSide,
770  const Point& aLastAtRemovalSide)
771 {
772  if (aFirstAtOppositeSide == aLastAtOppositeSide)
773  {
774  Vector newDirection = aFirstAtOppositeSide - aBezout;
775  Coordinate k; //number of repetition of newDirection
776 
777  Vector toLastAtRemovalSide = aLastAtRemovalSide - aFirst;
778  k = DGtal::ArithmeticalDSLKernel<TCoordinate,adjacency>::norm(toLastAtRemovalSide[1], toLastAtRemovalSide[0])
779  / DGtal::ArithmeticalDSLKernel<TCoordinate,adjacency>::norm(newDirection[1], newDirection[0]);
780  aFirstAtRemovalSide = aLastAtRemovalSide - newDirection*k;
781 
782  Vector toLast = aLast - aFirstAtOppositeSide;
783  k = DGtal::ArithmeticalDSLKernel<TCoordinate,adjacency>::norm(toLast[1], toLast[0])
784  / DGtal::ArithmeticalDSLKernel<TCoordinate,adjacency>::norm(newDirection[1], newDirection[0]);
785  aLastAtOppositeSide = aFirstAtOppositeSide + newDirection*k;
786  return true;
787  }
788  else
789  {
790  aFirstAtRemovalSide += aDirection;
791  return false;
792  }
793 }

821 template <typename TCoordinate, typename TInteger, unsigned short adjacency>
822 inline
823 typename DGtal::ArithmeticalDSS<TCoordinate, TInteger, adjacency>::PointD
824 DGtal::ArithmeticalDSS<TCoordinate, TInteger, adjacency>::
825 project( const Point& aM, double aR ) const
826 {
827  //casts
828  double aa = NumberTraits<Coordinate>::castToDouble(myDSL.myA);
829  double bb = NumberTraits<Coordinate>::castToDouble(myDSL.myB);
830  double xm = NumberTraits<Integer>::castToDouble(aM[0]);
831  double ym = NumberTraits<Integer>::castToDouble(aM[1]);
832 
833  //computation
834  double d2 = ( aa * aa + bb * bb );
835  double s = bb * xm + aa * ym;
836  double xp = ( bb * s + aa * aR ) / d2;
837  double yp = ( aa * s - bb * aR ) / d2;
838 
839  return PointD( xp, yp );
840 }
