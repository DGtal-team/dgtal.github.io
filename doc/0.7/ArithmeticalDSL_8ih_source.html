155 template <typename TCoordinate, typename TInteger, unsigned short adjacency>
156 inline
157 bool
158 DGtal::ArithmeticalDSL<TCoordinate, TInteger, adjacency>::isValid() const
159 {
160  bool flagIsValid; //'true' if valid, 'false' otherwise
161 
162  if ( (myA != 0) || (myB != 0) )
163  {
164  Integer gcd = IntegerComputer<Coordinate>::staticGcd(myA, myB);
165  if (gcd != NumberTraits<Coordinate>::ONE)
166  flagIsValid = false; //a,b must be relatively prime
167  else if (!checkShiftAndSteps())
168  flagIsValid = false; //redondant parameters must be consistent
169  else
170  flagIsValid = true;
171  }
172  else
173  flagIsValid = false; //a and b cannot be both null
174 
175  return flagIsValid;
176 }

178 //-----------------------------------------------------------------------------
179 template <typename TCoordinate, typename TInteger, unsigned short adjacency>
180 inline
181 bool
182 DGtal::ArithmeticalDSL<TCoordinate, TInteger, adjacency>::checkShiftAndSteps() const
183 {
184  ASSERT( (myA != 0) || (myB != 0) );
185 
186  bool flagIsValid; //'true' if valid, 'false' otherwise
187 
188  if (mySteps != DGtal::ArithmeticalDSLKernel<TCoordinate,adjacency>::steps(myA, myB))
189  flagIsValid = false;
190  else if (myShift != DGtal::ArithmeticalDSLKernel<TCoordinate,adjacency>::shift(myA, myB))
191  flagIsValid = false;
192  else if ( ( (mySteps.second[0] != NumberTraits<Coordinate>::ZERO)
193  ||(mySteps.second[1] != NumberTraits<Coordinate>::ZERO) )
194  &&( (mySteps.first - mySteps.second) != myShift ) )
195  flagIsValid = false;
196  else if ( remainder( myShift ) != (myUpperBound - myLowerBound + NumberTraits<Integer>::ONE) )
197  flagIsValid = false;
198  else
199  flagIsValid = true;
200 
201  return flagIsValid;
202 }
