207  typedef TInteger Integer;
208 
209  //get DSS parameters
210  Vector3d v; //direction vector
211  PointD3d mu; //intercept
212  PointD3d omega; //thickness
213  a.getParameters(v,mu,omega);
214 
215  //casting coordinates of v in double
216  VectorD3d u = VectorD3d(NumberTraits<Integer>::castToDouble(v[0]),
217  NumberTraits<Integer>::castToDouble(v[1]),
218  NumberTraits<Integer>::castToDouble(v[2]) );
219  //L2 norm of u
220  double n = u[0]*u[0] + u[1]*u[1] + u[2]*u[2];
221 
222  //first and last points
223  Point3d first = *a.begin();
224  Point3d last = *(--a.end());
225  PointD3d f = PointD3d(NumberTraits<Integer>::castToDouble(first[0]),
226  NumberTraits<Integer>::castToDouble(first[1]),
227  NumberTraits<Integer>::castToDouble(first[2]) );
228  PointD3d l = PointD3d(NumberTraits<Integer>::castToDouble(last[0]),
229  NumberTraits<Integer>::castToDouble(last[1]),
230  NumberTraits<Integer>::castToDouble(last[2]) );
231 
232  if (n != 0) {
233 
234  //last coefficient of the normal plane to the DSS direction
235  //passing trough f and l
236  double df = -u[0]*f[0] -u[1]*f[1] -u[2]*f[2];
237  double dl = -u[0]*l[0] -u[1]*l[1] -u[2]*l[2];
238 
239  //omega masks
240  PointD3d omega1, omega2;
241  if (omega[0] == 0) {
242  omega1 = PointD3d(0,omega[1],0);
243  omega2 = PointD3d(0,0,omega[2]);
244  } else {
245  if (omega[1] == 0) {
246  omega1 = PointD3d(omega[0],0,0);
247  omega2 = PointD3d(0,0,omega[2]);
248  } else {//omega[2] == 0
249  omega1 = PointD3d(omega[0],0,0);
250  omega2 = PointD3d(0,omega[1],0);
251  }
252  }
253 
254  double m1 = u[0]*mu[0] + u[1]*mu[1] + u[2]*mu[2];
255  double m2 = u[0]*(mu[0]+omega1[0]) + u[1]*(mu[1]+omega1[1]) + u[2]*(mu[2]+omega1[2]);
256  double m3 = u[0]*(mu[0]+omega2[0]) + u[1]*(mu[1]+omega2[1]) + u[2]*(mu[2]+omega2[2]);
257  double m4 = u[0]*(mu[0]+omega[0]) + u[1]*(mu[1]+omega[1]) + u[2]*(mu[2]+omega[2]);
258 
259  //4 lines
260  PointD3d pf = PointD3d( mu[0] - ( (m1+df)*u[0] )/n,
261  mu[1] - ( (m1+df)*u[1] )/n,
262  mu[2] - ( (m1+df)*u[2] )/n );
263  DGtal::Z3i::RealPoint rpf = display.embed( pf );
264  PointD3d pl = PointD3d( mu[0] - ( (m1+dl)*u[0] )/n,
265  mu[1] - ( (m1+dl)*u[1] )/n,
266  mu[2] - ( (m1+dl)*u[2] )/n );
267  DGtal::Z3i::RealPoint rpl = display.embed( pl );
268 
269  display.addLine(rpf, rpl);
270 
271  PointD3d pf2 = PointD3d((mu[0]+omega1[0]) - ( (m2+df)*u[0] )/n,
272  (mu[1]+omega1[1]) - ( (m2+df)*u[1] )/n,
273  (mu[2]+omega1[2]) - ( (m2+df)*u[2] )/n );
274  DGtal::Z3i::RealPoint rpf2 = display.embed( pf2 );
275  PointD3d pl2 = PointD3d((mu[0]+omega1[0]) - ( (m2+dl)*u[0] )/n,
276  (mu[1]+omega1[1]) - ( (m2+dl)*u[1] )/n,
277  (mu[2]+omega1[2]) - ( (m2+dl)*u[2] )/n );
278  DGtal::Z3i::RealPoint rpl2 = display.embed( pl2 );
279 
280  display.addLine(rpf2, rpl2);
281 
282  PointD3d pf3 = PointD3d((mu[0]+omega2[0]) - ( (m3+df)*u[0] )/n,
283  (mu[1]+omega2[1]) - ( (m3+df)*u[1] )/n,
284  (mu[2]+omega2[2]) - ( (m3+df)*u[2] )/n );
285  DGtal::Z3i::RealPoint rpf3 = display.embed( pf3 );
286  PointD3d pl3 = PointD3d((mu[0]+omega2[0]) - ( (m3+dl)*u[0] )/n,
287  (mu[1]+omega2[1]) - ( (m3+dl)*u[1] )/n,
288  (mu[2]+omega2[2]) - ( (m3+dl)*u[2] )/n );
289  DGtal::Z3i::RealPoint rpl3 = display.embed( pl3 );
290 
291  display.addLine(rpf3, rpl3);
292 
293  PointD3d pf4 = PointD3d((mu[0]+omega[0]) - ( (m4+df)*u[0] )/n,
294  (mu[1]+omega[1]) - ( (m4+df)*u[1] )/n,
295  (mu[2]+omega[2]) - ( (m4+df)*u[2] )/n );
296  DGtal::Z3i::RealPoint rpf4 = display.embed( pf4 );

299  (mu[2]+omega[2]) - ( (m4+dl)*u[2] )/n );

301 
302  display.addLine(rpf4, rpl4);
303 
304  //two end facets
