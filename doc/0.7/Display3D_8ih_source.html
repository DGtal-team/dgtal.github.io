607 template < typename Space ,typename KSpace >
608 inline
609 void
610 DGtal::Display3D< Space ,KSpace >::addCone(const RealPoint &p1, const RealPoint &p2,
611  double width)
612 {
613  updateBoundingBox(p1);
614  updateBoundingBox(p2);
615 
616  int nbPoints = 6; //the number of point on the base of the cone
617  double degree = 360 / nbPoints; // the angle between two points
618  double radian = degree * M_PI/180.0;
619  double norm; // norm of the vectors
620 
621  // A(x1,y1,z1) and B(x2,y2,z2) vector AB
622  double xab, yab, zab;
623 
624  xab = p2[0]-p1[0];
625  yab = p2[1]-p1[1];
626  zab = p2[2]-p1[2];
627 
628  norm = sqrt( xab*xab + yab*yab + zab*zab);
629  assert (norm > 0);
630  xab = xab / norm;
631  yab = yab / norm;
632  zab = zab / norm;
633 
634  //take a third point M different from A and B
635  double xm = 0;
636  double ym = 0;
637  double zm = 0;
638 
639  while (p1[0] == xm || p2[0] ==xm) xm++;
640  while (p1[1] == ym || p2[1] ==ym) ym++;
641  while (p1[2] == zm || p2[2] ==zm) zm++;
642 
643  //vector AM
644  double xam, yam, zam;
645 
646  xam = xm-p1[0];
647  yam = ym-p1[1];
648  zam = zm-p1[2];
649 
650  norm = sqrt( xam*xam + yam*yam + zam*zam);
651  assert (norm > 0);
652  xam = xam / norm;
653  yam = yam / norm;
654  zam = zam / norm;
655 
656  //normal vector n = AB^AM
657  double xn, yn, zn;
658 
659  xn = yab*zam - yam*zab;
660  yn = xam*zab - xab*zam;
661  zn = xab*yam - xam*yab;
662 
663  //divide n by its norm
664  norm = sqrt( xn*xn + yn*yn + zn*zn);
665  assert (norm > 0);
666  xn = xn / norm;
667  yn = yn / norm;
668  zn = zn / norm;
669 
670  //the first point of the base
671  double xf, yf, zf;
672 
673  xf = p2[0] + width * xn;
674  yf = p2[1] + width * yn;
675  zf = p2[2] + width * zn;
676 
677  //two following points of the base
678  double xb1 = xf;
679  double yb1 = yf;
680  double zb1 = zf;
681  double xb2,yb2, zb2;
682 
683  //r = AB^n for the future rotation
684  double xr, yr, zr;
685 
686  createNewTriangleList("Cone");
687  for (int i =0; i < nbPoints-1; i ++)
688  {
689  // r = AB^n
690  xr = yab*zn - yn*zab;
691  yr = xn*zab - xab*zn;
692  zr = xab*yn - xn*yab;
693 
694  //rotate n by degree
695  xn = cos(radian)*xn + sin(radian)*xr;
696  yn = cos(radian)*yn + sin(radian)*yr;
697  zn = cos(radian)*zn + sin(radian)*zr;
698 
699  //divide n by its norm
700  norm = sqrt( xn*xn + yn*yn + zn*zn);
701  assert (norm > 0);
702  xn = xn / norm;
703  yn = yn / norm;
704  zn = zn / norm;
705 
706  // calculate one point with the normal vector at a distance width
707  xb2 = p2[0] + width * xn;
708  yb2 = p2[1] + width * yn;
709  zb2 = p2[2] + width * zn;
710 
711  //adding the triangles associated with this point
712  addTriangle(p1,
713  RealPoint(xb1, yb1, zb1),
714  RealPoint(xb2, yb2, zb2));
715  addTriangle(p2,
716  RealPoint(xb1, yb1, zb1),
717  RealPoint(xb2, yb2, zb2));
718 
719  xb1 = xb2;
720  yb1 = yb2;
721  zb1 = zb2;
722  }
723 
724  //Last part to close the cone
725  addTriangle(p1,
726  RealPoint(xb1, yb1,zb1),
727  RealPoint(xf, yf, zf));
728  addTriangle(p2,
729  RealPoint(xb1, yb1, zb1),
730  RealPoint(xf, yf, zf));
731 }
