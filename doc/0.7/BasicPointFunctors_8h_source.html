272  Point operator()(const TPointDimMinus& aPoint) const
273  {
274  Point pt;
275  Dimension pos=0;
276  std::vector<Dimension> indexesRotate;
277  for( Dimension i=0; i<pt.size(); i++)
278  {
279  if(i!=myPosDimAdded)
280  {
281  pt[i]= aPoint[pos];
282  pos++;
283  }else
284  {
285  pt[i]=mySliceIndex;
286  }
287  }
288  for( Dimension i=0; i<pt.size(); i++)
289  {
290  if(i!=myDimRotated)
291  indexesRotate.push_back(i);
292  }
293  double d1 = pt[indexesRotate[0]] - myCenter[indexesRotate[0]];
294  double d2 = pt[indexesRotate[1]] - myCenter[indexesRotate[1]];
295 
296  pt[indexesRotate[0]] = myCenter[indexesRotate[0]] + d1*cos(myRotationAngle)-d2*sin(myRotationAngle) ;
297  pt[indexesRotate[1]] = myCenter[indexesRotate[1]] + d1*sin(myRotationAngle)+d2*cos(myRotationAngle) ;
298 
299  if(myDomain.isInside(pt))
300  return pt;
301  else
302  return myDefaultPoint;
303  }

404  double d = -anNormalVector[0]*anOriginPoint[0] - anNormalVector[1]*anOriginPoint[1] - anNormalVector[2]*anOriginPoint[2];
405  typename Space::RealPoint pRefOrigin;

411  pRefOrigin[1]=-1.0;

416  pRefOrigin [2]= 0.0;
417  if(pRefOrigin==anOriginPoint){
418  pRefOrigin[0]=-1.0;
419  }
420  }else if (anNormalVector[2]!=0){
421  pRefOrigin [0]= 0.0;
422  pRefOrigin [1]= 0.0;
423  pRefOrigin [2]= -d/anNormalVector[2];
424  if(pRefOrigin==anOriginPoint){
425  pRefOrigin[0]=-1.0;
426  }
427  }

429  uDir1=(pRefOrigin-anOriginPoint)/((pRefOrigin-anOriginPoint).norm());
