100  }

102  {
103  theta=atan2(NumberTraits<typename T::Component>::castToDouble(aDir[1]),
104  NumberTraits<typename T::Component>::castToDouble(aDir[0]));
105 
106  if(aDir[1]<0)
107  theta += 2.0*M_PI;
108 
109  Nthetai = floor(2.0*(myNphi)*sin(posPhi*dphi));
