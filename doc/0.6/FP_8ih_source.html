65 inline
66 void
67 DGtal::FP<TIterator,TInteger,connectivity>::
68 algorithm(const TIterator& itb,
69  const TIterator& ite, IteratorType ) throw( InputException )
70 {
71  ASSERT(isNotEmpty(itb,ite));
72 
74  //list of successive upper (U) and lower (L)
75  // leaning points.
76  std::list<Point> vTmpU, vTmpL;
77  vTmpU.push_back(*itb);
78  vTmpL.push_back(*itb);
79 
80  //longest DSS
81  DSSComputer longestDSS;
82  longestDSS.init(itb);
83 
84  while ( (longestDSS.end() != ite)&&(longestDSS.extendForward()) )
85  {
86  //store the first upper leaning point
87  if (longestDSS.getUf() != vTmpU.back()) {
88  vTmpU.push_back(longestDSS.getUf());
89  }
90  //store the first lower leaning point
91  if (longestDSS.getLf() != vTmpL.back()) {
92  vTmpL.push_back(longestDSS.getLf());
93  }
94  }
95 
96  //std::cerr << "First MS" << std::endl << longestDSS << std::endl;
97 
98  typedef details::DSSDecorator<DSSComputer> DSSDecorator;
99  DSSDecorator* adapter = 0; //adapter for the current DSS
100 
101  if (longestDSS.end() != ite)
102  {
103  //if the curve is not straight
104  //it is either locally convex or concave at longestDSS.end()
105  adapter = initConvexityConcavity<DSSDecorator>( longestDSS );
106 
107  ASSERT(adapter);
108  if ( adapter->isInConvexPart() ) myPolygon = vTmpU;
109  else myPolygon = vTmpL;
110  ASSERT(myPolygon.size() > 0);
111 
112  bool go = true;
113  while ( go )
114  { //main loop
115  if ( !removingStep( adapter ) )
116  { //disconnected digital curve
117  throw InputException();
118  }
119  go = addingStep( adapter, ite );
120  if ( go )
121  {
122  delete (adapter);
123  adapter = initConvexityConcavity<DSSDecorator>( longestDSS );
124  }
125  }
126 
127  }
128  else
129  {
130  //the curve is assumed to be convex
131  //if it is straight
132  myPolygon = vTmpU;
133  adapter = new details::DSSDecorator4ConvexPart<DSSComputer>(longestDSS);
134  ASSERT(adapter);
135  }
136 
137  //store the last leaning point
138  //if the first and last leaning points
139  //of the MS are not confounded
140  if (adapter->firstLeaningPoint() != adapter->lastLeaningPoint()) {
141  myPolygon.push_back(adapter->lastLeaningPoint());
142  }
143 
144  //last removing step
145  while ( longestDSS.retractForward() ) {
146  //store the last leaning point
147  if (adapter->lastLeaningPoint() != myPolygon.back()) {
148  myPolygon.push_back(adapter->lastLeaningPoint());
149  }
150  }
151 
152  ASSERT(adapter);//had to be allocated
153  delete (adapter);
154 }

158 void
159 DGtal::FP<TIterator,TInteger,connectivity>::
160 algorithm(const TIterator& itb,
161  const TIterator& ite, CirculatorType ) throw( InputException )
162 {
163  ASSERT(isNotEmpty(itb,ite));
164 
166  //first maximal DSS
167  DSSComputer currentDSS;
168  currentDSS.init( itb );
169  //forward extension
170  while ( currentDSS.extendForward() ) {}
171  //backward extension
172  while ( currentDSS.extendBackward() ) {}
173 
174  //std::cerr << "First MS" << std::endl << currentDSS << std::endl;
175 
176  //local convexity
177  typedef details::DSSDecorator<DSSComputer> DSSDecorator;
178  DSSDecorator* adapter = 0; //adapter for the current DSS
179  adapter = initConvexityConcavity<DSSDecorator>( currentDSS );
180 
181  //stopping point
182  Point stoppingPoint = adapter->lastLeaningPoint();
183  if (adapter->firstLeaningPoint() == stoppingPoint) {
184  myPolygon.push_back( stoppingPoint );
185  }//stored in removingStep function otherwise
186 
187  //since there are more than 2 MS (closed case)
188  //but no more than 2 MS sharing a leaning point
189  //at the same position
190  if ( !removingStep( adapter ) )
191  { //disconnected digital curve
192  throw InputException();
193  }
194  addingStep( adapter );
195  bool go = true;
196  while ( go )
197  { //main loop
198  delete (adapter);
199  adapter = initConvexityConcavity<DSSDecorator>( currentDSS );
200 
201  if ( !removingStep( adapter ) )
202  { //disconnected digital curve
203  throw InputException();
204  }
205  addingStep( adapter );
206  if ( adapter->lastLeaningPoint() == stoppingPoint )
207  { //stop and remove the last point to avoid a repetition
208  if (adapter->firstLeaningPoint() == stoppingPoint) {
209  ASSERT( myPolygon.size() > 0 );
210  myPolygon.pop_back();
211  }
212  go = false;
213  }
214  }
215 
216  delete (adapter);
217 }

391 inline
392 bool
393 DGtal::FP<TIterator,TInteger,connectivity>::isValid() const
394 {
395  typedef typename Polygon::const_iterator I;
396 
397  //not two consecutive confounded points
398  bool flag1 = true;
399  I it = myPolygon.begin();
400  if (it != myPolygon.end())
401  {
402  I itPrev = it; ++it;
403  if (it != myPolygon.end())
404  {
405  while ( (it != myPolygon.end())&&(flag1) )
406  {
407  flag1 = (*itPrev != *it);
408  itPrev = it;
409  ++it;
410  }
411  }//if only one point ok
412  }//if no point ok
413 
414  //valid quadrant for all consecutive pairs of edges
415  bool flag2 = true;
416  if (myPolygon.size() >= 3) {
417 
418  I i, j, k;
419  i = myPolygon.begin();
420  j = i; ++j;
421  k = j; ++k;
422 
423  if ( isClosed() ) {
424 
425  //first point
426  flag2 = isValid(myPolygon.back(),*i,*j);
427  //middle points
428  while ( (k != myPolygon.end())&&(flag2) ) {
429  flag2 = isValid(*i,*j,*k);
430  ++i; ++j; ++k;
431  }
432  //last point
433  if (flag2)
434  flag2 = isValid(*i,*j,myPolygon.front());
435 
436  } else {
437 
438  //middle points
439  while ( (k != myPolygon.end())&&(flag2) ) {
440  flag2 = isValid(*i,*j,*k);
441  ++i; ++j; ++k;
442  }
443 
444  }
445  }//special case of less than 3 points ok
446 
447  return flag1 && flag2;
448 }

525 inline
526 DGtal::PointVector<2,double>
527 DGtal::FP<TIterator,TInteger,connectivity>
528 ::getRealPoint (const Point& a,const Point& b, const Point& c) const {
529 
530  ASSERT( isValid() );
531 
532  RealVector shift;
533 
534  Vector e1 = b - a; //previous edge
535  Vector e2 = c - b; //next edge
536 
537  if ( (e1[0]*e2[1]-e1[1]*e2[0]) <= 0 ) {
538 
539  //convex turn
540  if ( (quadrant(e1,1))&&(quadrant(e2,1)) ) {
541  shift = RealVector(0.5,-0.5);
542  } else if ( (quadrant(e1,2))&&(quadrant(e2,2)) ) {
543  shift = RealVector(-0.5,-0.5);
544  } else if ( (quadrant(e1,3))&&(quadrant(e2,3)) ) {
545  shift = RealVector(-0.5,0.5);
546  } else if ( (quadrant(e1,4))&&(quadrant(e2,4)) ) {
547  shift = RealVector(0.5,0.5);
548  } else {
549  ASSERT(false && "DGtal::FP<TIterator,TInteger,connectivity>::getRealPoint: not valid polygon" );
550  }
551 
552  } else {
553 
554  //concave turn
555  if ( (quadrant(e1,1))&&(quadrant(e2,1)) ) {
556  shift = RealVector(-0.5,0.5);
557  } else if ( (quadrant(e1,2))&&(quadrant(e2,2)) ) {
558  shift = RealVector(0.5,0.5);
559  } else if ( (quadrant(e1,3))&&(quadrant(e2,3)) ) {
560  shift = RealVector(0.5,-0.5);
561  } else if ( (quadrant(e1,4))&&(quadrant(e2,4)) ) {
562  shift = RealVector(-0.5,-0.5);
563  } else {
564  ASSERT(false && "DGtal::FP<TIterator,TInteger,connectivity>::getRealPoint: not valid polygon" );
565  }
566 
567  }
568 
569  return ( RealPoint(b) + shift );
