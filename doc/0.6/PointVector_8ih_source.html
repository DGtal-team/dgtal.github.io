641 //------------------------------------------------------------------------------
642 template<DGtal::Dimension dim, typename TComponent, typename TContainer>
643 inline
644 double
645 DGtal::PointVector<dim, TComponent, TContainer>::norm (const
646  typename Self::NormType aType ) const
647 {
648  double tmp = 0.0;
649 
650  ASSERT ( dim > 0 );
651 
652  switch ( aType )
653  {
654  case L_2:
655  for ( DGtal::Dimension i = 0; i < dimension; i++ )
656  tmp += NumberTraits<Component>::castToDouble(myArray[ i ]) *
657  NumberTraits<Component>::castToDouble(myArray[ i ]);
658  tmp = ( double ) sqrt ( tmp );
659  break;
660  case L_1:
661  for ( DGtal::Dimension i = 0; i < dimension; i++ )
662  tmp += fabs ( NumberTraits<Component>::castToDouble(myArray[ i ] ));
663  break;
664  case L_infty:
665  tmp = fabs( NumberTraits<Component>::castToDouble( myArray[ 0 ]));
666  for ( DGtal::Dimension i = 1; i < dimension; i++ )
667  if ( tmp < fabs ( NumberTraits<Component>::castToDouble(myArray[ i ] ) ))
668  tmp = fabs ( NumberTraits<Component>::castToDouble(myArray[ i ] ));
