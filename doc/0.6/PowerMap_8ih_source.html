100 inline
101 void
102 DGtal::PowerMap<W, Sep,Im>::computeOtherSteps ( const Dimension dim ) const
103 {
104  std::string title = "Powermap dimension " + boost::lexical_cast<std::string>( dim ) ;
105  trace.beginBlock ( title );
106 
107  typedef typename Domain::ConstSubRange::ConstIterator ConstDomIt;
108 
109  //We setup the subdomain iterator
110  //the iterator will scan dimension using the order:
111  // {n-1, n-2, ... 1} (we skip the '0' dimension).
112  std::vector<Size> subdomain;
113  subdomain.reserve(W::Domain::Space::dimension - 1);
114  for (unsigned int k = 0; k < W::Domain::Space::dimension ; k++)
115  if ( ((int)W::Domain::Space::dimension - 1 - k) != dim)
116  subdomain.push_back( (int)W::Domain::Space::dimension - 1 - k );
117 
118  Domain localDomain(myLowerBoundCopy, myUpperBoundCopy);
119 
120 
121 #ifdef WITH_OPENMP
122  //Parallel loop
123  std::vector<Point> subRangePoints;
124  //Starting point precomputation
125  for (ConstDomIt it = localDomain.subRange( subdomain ).begin(),
126  itend = localDomain.subRange( subdomain ).end();
127  it != itend; ++it)
128  subRangePoints.push_back( *it );
129  //We run the 1D problems in //
130 #pragma omp parallel for schedule(dynamic)
131  for (size_t i = 0; i < subRangePoints.size(); ++i)
132  computeOtherStep1D ( subRangePoints[i], dim);
133 
134 #else
135  //We solve the 1D problems sequentially
136  for (ConstDomIt it = localDomain.subRange( subdomain ).begin(),
137  itend = localDomain.subRange( subdomain ).end();
138  it != itend; ++it)
139  computeOtherStep1D ( *it, dim);
140 #endif
141 
142  trace.endBlock();
143 
144 }

148 template <typename W, typename Sep, typename Im>
149 void
150 DGtal::PowerMap<W,Sep,Im>::computeOtherStep1D ( const Point &startingPoint,
151  const Size dim) const
152 {
153  Point point = startingPoint;
154  Point endpoint = startingPoint;
155  Point psite;
156  int nbSites = -1;
157  std::vector<Point> Sites;
158 
159  //Reserve
160  Sites.reserve( myUpperBoundCopy[dim] - myLowerBoundCopy[dim] +1);
161 
162  //endpoint of the 1D row
163  endpoint[dim] = myUpperBoundCopy[dim];
164 
165  //Pruning the list of sites (dim=0 implies no hibben sites)
166  if (dim==0)
167  {
168  for(Abscissa i = myLowerBoundCopy[dim] ; i <= myUpperBoundCopy[dim] ; i++)
169  {
170  psite = myImagePtr->operator()( point );
171  if ( psite != myInfinity )
172  {
173  nbSites++;
174  Sites.push_back( psite );
175  }
176  point[dim] ++;
177  }
178  }
179  else
180  {
181  //Pruning the list of sites
182  for(Abscissa i = myLowerBoundCopy[dim] ; i <= myUpperBoundCopy[dim] ; i++)
183  {
184  psite = myImagePtr->operator()(point);
185  if ( psite != myInfinity )
186  {
187  while ((nbSites >= 1) &&
188  ( myMetricPtr->hiddenByPower(Sites[nbSites-1], myWeightImagePtr->operator()(Sites[nbSites-1]),
189  Sites[nbSites] , myWeightImagePtr->operator()(Sites[nbSites]),
190  psite, myWeightImagePtr->operator()(psite),
191  startingPoint, endpoint, dim) ))
192  {
193  nbSites --;
194  Sites.pop_back();
195  }
196  nbSites++;
197  Sites.push_back( psite );
198  }
199  point[dim] ++;
200  }
201  }
202 
203 
204  if (startingPoint[0] == 4)
205  {
206  for(typename std::vector<Point>::const_iterator it = Sites.begin(), itend=Sites.end(); it!=itend; ++it)
207  trace.info() << *it;
208  trace.info()<<std::endl;
209  }
210  //No sites found
211  if (nbSites == -1)
212  return;
213 
214  int k = 0;
215 
216  //Rewriting
217  point[dim] = myLowerBoundCopy[dim];
218  for(Abscissa i = myLowerBoundCopy[dim] ; i <= myUpperBoundCopy[dim] ; i++)
219  {
220  while ( (k < nbSites) &&
221  ( myMetricPtr->closestPower(point,
222  Sites[k], myWeightImagePtr->operator()(Sites[k]),
223  Sites[k+1],myWeightImagePtr->operator()(Sites[k+1]))
224  != DGtal::ClosestFIRST ))
225  k++;
226 
227  myImagePtr->setValue(point, Sites[k]);
228  point[dim]++;
229  }
230 }
