60TriMesh makeTwoTriangles()
61{
62 TriMesh mesh;

64 mesh.addVertex( RealPoint( 1, 0, 0 ) );

185 GIVEN( "Two triangles incident by an edge" ) {

187 auto nbv = trimesh.nbVertices();
188 auto nbe = trimesh.nbEdges();
189 auto nbf = trimesh.nbFaces();
190 int nbfl = 0;
191 ArcT afl = 0;

195 afl = a;
196 }
