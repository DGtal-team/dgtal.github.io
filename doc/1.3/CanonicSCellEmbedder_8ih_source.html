92typename DGtal::CanonicSCellEmbedder<TKSpace>::RealPoint
93DGtal::CanonicSCellEmbedder<TKSpace>::embed( const SCell & scell ) const

95 return this->operator()( scell );

97//-----------------------------------------------------------------------------
98template <typename TKSpace>
99inline
100typename DGtal::CanonicSCellEmbedder<TKSpace>::RealPoint
101DGtal::CanonicSCellEmbedder<TKSpace>::operator()( const SCell & scell ) const
102{
103 ASSERT( this->isValid() );
104 Point dp = myKSpace->sKCoords( scell );
105 RealPoint p;
106 for ( Dimension i = 0; i < dp.size(); ++i )
107 p[ i ] = ( NumberTraits<Integer>::castToDouble( dp[ i ] ) / 2.0 ) - 0.5; /// In Khalimsky Space, 0-SCells have even coordinates, while in the embedding, voxel centers have integer coordinates, we need to shift all coordinates by -1/2.
108 return p;
109}

117}
