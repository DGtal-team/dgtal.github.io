235 template <typename SC>
236 bool maximalSymmetricExtension(SC& s,
237  const typename SC::ConstIterator& begin,
238  const typename SC::ConstIterator& end,
239  IteratorType ) {
240 
241  bool flagOk = true;
242  bool flagForward = true;
243  //while the extension is possible
244  //at the front and (then) at the back
245  while (flagOk) {
246  if (flagForward) {
247  flagForward = false;
248  if ( s.end() != end ) flagOk = s.extendFront();
249  else flagOk = false;
250  } else {
251  flagForward = true;
252  if ( s.begin() != begin ) flagOk = s.extendBack();
253  else flagOk = false;
254  }
255  }
256  //extend one more time if s.begin() == begin
257  if (s.begin() != begin ) {
258  if (s.extendBack()) return !s.extendFront();
259  else return false;
260  } else {
261  return !flagForward;
262  }
263 
264 }

516 template <typename SC>
517 void mostCenteredMaximalSegment(SC& s,
518  const typename SC::ConstIterator& i,
519  const typename SC::ConstIterator& begin,
520  const typename SC::ConstIterator& end,
521  DGtal::ForwardSegmentComputer )
522 {
523 
524  typedef typename SC::ConstIterator ConstIterator;
525  typedef typename SC::Reverse ReverseSegmentComputer;
526  typedef typename ReverseSegmentComputer::ConstIterator ConstReverseIterator;
527 
528  //get the first maximal segment passing through i
529 
530  firstMaximalSegment( s, i, begin, end, DGtal::ForwardSegmentComputer() );
531 
532  //get the next maximal segment while i is not at the middle of
533  //the current maximal segment.
534 
535  ConstIterator k( s.begin() );
536  while ( k != i ) {
537 
538  if ( isNotEmpty<ConstIterator>(s.end(),end) ) {
539 
540  //backward extension
541  ConstIterator it( s.end() ); ++it;
542  ConstReverseIterator rit( it );
543  ConstReverseIterator rend( s.begin() );
544  ReverseSegmentComputer r( s.getReverse() );
545  longestSegment(r, rit, rend);
546  ConstIterator newBegin = r.end().base();
547  ASSERT( newBegin != s.begin() );
548 
549  while ( ( k != getMiddleIterator(newBegin, s.end() ) )
550  &&( k != i ) ) {
551  ++k;
552  }
553  if ( k != i ) {
554 
555  //get the next maximal segment
556  longestSegment(s, newBegin, end);
557 
558  }
559 
560  } else {
561  k = i;
562  }
563  }
564 }
