63 struct ImplicitDigitalEllipse3 {
64  typedef TPoint3 Point;

70  bool operator()( const TPoint3 & p ) const
71  {
72  double x = ( (double) p[ 0 ] / myA );
73  double y = ( (double) p[ 1 ] / myB );
74  double z = ( (double) p[ 2 ] / myC );
