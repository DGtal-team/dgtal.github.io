85double CalculatePi(int depth) {
86 double pi = 0.0;
87 for (int i = 0; i < depth; ++i) {
88 double numerator = static_cast<double>(((i % 2) * 2) - 1);
89 double denominator = static_cast<double>((2 * i) - 1);
90 pi += numerator / denominator;
91 }
92 return (pi - 1.0) * 4;
93}
