30 double standard_deviation(const VectorXd& xx)
31 {
32  const double mean = xx.mean();
33  double accum = 0;
34  for (int kk=0, kk_max=xx.rows(); kk<kk_max; kk++)
35  accum += (xx(kk)-mean)*(xx(kk)-mean);
36  return sqrt(accum)/(xx.size()-1);
37 }
