150  // Full domain
151  auto const domain = anImage.domain();
152 
153  // Sub domain
154  Point lowerPt = domain.lowerBound();
155  Point upperPt = domain.upperBound();
156  for ( Dimension i = 0; i < Domain::dimension; ++i )
157  {
