39 template <typename Image>
40 template <typename Set>
41 inline
42 Image
43 DGtal::ImageFromSet<Image>::create(const Set &aSet, const Value &defaultValue,
44  const bool addBorder,
45  typename Set::ConstIterator itBegin,
46  typename Set::ConstIterator itEnd)
47 {
48  BOOST_CONCEPT_ASSERT(( concepts::CDigitalSet<Set> ));
49 
50  typename Set::Point lower,upper;
51  typename Set::Point dec;
52 
53  if (addBorder)
54  dec = Set::Point::diagonal(1);
55 
56  aSet.computeBoundingBox(lower,upper);
57 
58  Image image(typename Image::Domain(lower - dec ,upper + dec));
59  append<Set>(image,defaultValue, itBegin, itEnd);
60 
61  return image;
62 
63 }
