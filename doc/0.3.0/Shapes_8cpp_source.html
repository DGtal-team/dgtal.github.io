00804 Shape &
00805 Ellipse::scale( double sx, double sy )
00806 {
00807   // Thanks to Freddie Exall for pointing an error with the first version
00808   // of this function, and for pointing to a fix as well!
00809   if ( _angle != 0 ) {
00810     double co = cos( _angle );
00811     double si = sin( _angle );
00812 
00813     // current transformation matrix
00814     double m00 = ( 1 / _xRadius ) * co;
00815     double m01 = ( 1 / _xRadius ) * si;
00816     double m10 = - ( 1 / _yRadius ) * si;
00817     double m11 = ( 1 / _yRadius ) * co;
00818 
00819     // Implicit function of ellipse at current
00820     // ax^2 + bxy + cy^2 = 1
00821     double a = ( m00 * m00 ) + ( m10 * m10 );
00822     double b = 2 * ( ( m00 * m01 ) + ( m10 * m11 ) );
00823     double c = ( m01 * m01 ) + ( m11 * m11 );
00824 
00825     // Scale coefficients ( x_new = sx * x, y_new = sy * y )
00826     a = a / ( sx * sx );
00827     b = b / ( sx * sy );
00828     c = c / ( sy * sy );
00829 
00830     if ( b == 0 ) {
00831       _angle = 0;
00832     } else if ( a == c ) {
00833       _angle = M_PI / 4;
00834       a += ( b / 2 );
00835       c -= ( b / 2 );
00836     } else {
00837       _angle = 0.5 * atan( b / ( a - c ) );
00838       double k = 1 + ( ( b * b ) / ( ( a - c ) * ( a - c ) ) );
00839       k = sqrt( k );
00840       k *= ( a - c );
00841       c += a;
00842       a = 0.5 * ( c + k );
00843       c = 0.5 * ( c - k );
00844     }
00845     _xRadius = 1 / sqrt( a );
00846     _yRadius = 1 / sqrt( c );
00847   } else {
00848     _xRadius = _xRadius * sx;
00849     _yRadius = _yRadius * sy;
00850   }
00851   return *this;
00852 }
