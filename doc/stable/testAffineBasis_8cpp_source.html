48template < typename Point >
49std::vector< Point >
50makeRandomVectors( int nb, int amplitude )
51{
52 std::uniform_int_distribution<int> U(-amplitude, amplitude);
53 std::vector< Point > P;
54 for ( auto n = 0; n < nb; ++n )
55 {
56 Point A;
57 for ( auto i = 0; i < Point::dimension; i++ )
58 A[ i ] = U( g );
59 P.push_back( A );
60 }
61 return P;
62}

64template < typename Point >
65std::vector< Point >
66makeRandomLatticePointsFromDirVectors( int nb, const vector< Point>& V )
67{
68 std::uniform_int_distribution<int> U(-10, 10);
69 vector< Point > P;
70 int n = V[0].size();
71 int m = V.size();
72 Point A;
73 for ( auto i = 0; i < n; i++ )
74 A[ i ] = U( g );
75 P.push_back( A );
76 for ( auto k = 0; k < nb; k++ )
77 {
78 Point B = A;
79 for ( auto i = 0; i < m; i++ )
80 {
81 int l = U( g );
82 B += l * V[ i ];
83 }
84 P.push_back( B );
85 }
86 std::shuffle( P.begin(), P.end(), g );
87 return P;
88}

90template <typename TPoint>
91bool sameDirection( const TPoint& a, const TPoint& b )
92{
93 return ( a[ 0 ] == b[ 0 ] ) ? ( a == b ) : ( a == -b );
94}

96template <typename TPoint>
97std::pair<TPoint, TPoint>
98boundingBox( const std::vector< TPoint >& v )
99{
100 TPoint l, u;
101 if ( ! v.empty() )
102 {
103 l = u = v[ 0 ];
104 for ( auto i = 1; i < v.size(); i++ )
105 {
106 l = l.inf( v[ i ] );
107 u = u.sup( v[ i ] );
108 }
109 }
110 return std::make_pair( l, u );
111}

113template <typename T, typename C>
114T
115determinant( const PointVector<2, T, C>& u,
116 const PointVector<2, T, C>& v )
117{
118 return u [ 0 ] * v [ 1 ] - u [ 1 ] * v [ 0 ];
119}
120
121template <typename T, typename C>
122T
123determinant( const PointVector<3, T, C>& u,
124 const PointVector<3, T, C>& v,
125 const PointVector<3, T, C>& w )
126{
127 typedef PointVector<2,T,C> P2;
128 return u[ 0 ] * determinant( P2( v[1], v[2] ), P2( w[1], w[2] ) )
129 - u[ 1 ] * determinant( P2( u[0], u[2] ), P2( w[0], w[2] ) )
130 + u[ 2 ] * determinant( P2( u[0], u[1] ), P2( v[0], v[1] ) );
131}
