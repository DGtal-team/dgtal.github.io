380  /* circle_circle_intersection() *
381  * Determine the points where 2 circles in a common plane intersect.
382  *
383  * int circle_circle_intersection(
384  * // center and radius of 1st circle
385  * double x0, double y0, double r0,
386  * // center and radius of 2nd circle
387  * double x1, double y1, double r1,
388  * // 1st intersection point
389  * double *xi, double *yi,
390  * // 2nd intersection point
391  * double *xi_prime, double *yi_prime)
392  *
393  * This is a public domain work. 3/26/2005 Tim Voght
394  *
395  */
403  static int circle_circle_intersection(double x0, double y0, double r0,
404  double x1, double y1, double r1,
405  double *xi, double *yi,
406  double *xi_prime, double *yi_prime)
407  {
408  double a, dx, dy, d, h, rx, ry;
409  double x2, y2;
410 
411  /* dx and dy are the vertical and horizontal distances between
412  * the circle centers.
413  */
414  dx = x1 - x0;
415  dy = y1 - y0;
416 
417  /* Determine the straight-line distance between the centers. */
418  //d = sqrt((dy*dy) + (dx*dx));
419  d = hypot(dx,dy); // Suggested by Keith Briggs
420 
421  /* Check for solvability. */
422  if (d > (r0 + r1))
423  {
424  /* no solution. circles do not intersect. */
425  std::cerr << "Warning : the two circles do not intersect -> should never happen" << std::endl;
426  return 0; //I. Sivignon 05/2010 should never happen for our specific use.
427  }
428  if (d < fabs(r0 - r1))
429  {
430  /* no solution. one circle is contained in the other */
431  return 0;
432  }
433 
434  /* 'point 2' is the point where the line through the circle
435  * intersection points crosses the line between the circle
436  * centers.
437  */
438 
439  /* Determine the distance from point 0 to point 2. */
440  a = ((r0*r0) - (r1*r1) + (d*d)) / (2.0 * d) ;
441 
442  /* Determine the coordinates of point 2. */
443  x2 = x0 + (dx * a/d);
444  y2 = y0 + (dy * a/d);
445 
446  /* Determine the distance from point 2 to either of the
447  * intersection points.
448  */
449  h = sqrt((r0*r0) - (a*a));
450 
451  /* Now determine the offsets of the intersection points from
452  * point 2.
453  */
454  rx = -dy * (h/d);
455  ry = dx * (h/d);
456 
457  /* Determine the absolute intersection points. */
458  *xi = x2 + rx;
459  *xi_prime = x2 - rx;
460  *yi = y2 + ry;
461  *yi_prime = y2 - ry;
462 
463  return 1;
464  }
