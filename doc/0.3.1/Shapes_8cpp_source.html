00999 Shape &
01000 Ellipse::scale( double sx, double sy )
01001 {
01002   // Thanks to Freddie Exall for pointing an error with the first version
01003   // of this function, and for pointing to a fix as well!
01004   if ( _angle != 0 ) {
01005     double co = cos( _angle );
01006     double si = sin( _angle );
01007 
01008     // current transformation matrix
01009     double m00 = ( 1 / _xRadius ) * co;
01010     double m01 = ( 1 / _xRadius ) * si;
01011     double m10 = - ( 1 / _yRadius ) * si;
01012     double m11 = ( 1 / _yRadius ) * co;
01013 
01014     // Implicit function of ellipse at current
01015     // ax^2 + bxy + cy^2 = 1
01016     double a = ( m00 * m00 ) + ( m10 * m10 );
01017     double b = 2 * ( ( m00 * m01 ) + ( m10 * m11 ) );
01018     double c = ( m01 * m01 ) + ( m11 * m11 );
01019 
01020     // Scale coefficients ( x_new = sx * x, y_new = sy * y )
01021     a = a / ( sx * sx );
01022     b = b / ( sx * sy );
01023     c = c / ( sy * sy );
01024 
01025     if ( b == 0 ) {
01026       _angle = 0;
01027     } else if ( a == c ) {
01028       _angle = M_PI / 4;
01029       a += ( b / 2 );
01030       c -= ( b / 2 );
01031     } else {
01032       _angle = 0.5 * atan( b / ( a - c ) );
01033       double k = 1 + ( ( b * b ) / ( ( a - c ) * ( a - c ) ) );
01034       k = sqrt( k );
01035       k *= ( a - c );
01036       c += a;
01037       a = 0.5 * ( c + k );
01038       c = 0.5 * ( c - k );
01039     }
01040     _xRadius = 1 / sqrt( a );
01041     _yRadius = 1 / sqrt( c );
01042   } else {
01043     _xRadius = _xRadius * sx;
01044     _yRadius = _yRadius * sy;
01045   }
01046   return *this;
01047 }
