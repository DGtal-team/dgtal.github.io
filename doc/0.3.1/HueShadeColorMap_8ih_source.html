00184 void
00185 DGtal::HueShadeColorMap<Value,DefaultCycles>::HSVtoRGB
00186 ( double & r, double & g, double & b,
00187   const double h, const double s, const double v)
00188 {
00189   int i;
00190   double f, p, q, t;
00191   if( s == 0 ) {                     // achromatic (gray)
00192     r = g = b = v;
00193     return;
00194   }
00195   i = static_cast<int>( floor( h / 60 ) );
00196   f = ( h / 60 ) - i;                        // factorial part of h
00197   p = v * ( 1.0 - s );
00198   q = v * ( 1.0 - s * f );
00199   t = v * ( 1.0 - s * ( 1.0 - f ) );
00200   switch( i ) {
00201   case 0:
00202     r = v; g = t; b = p;
00203     break;
00204   case 1:
00205     r = q; g = v; b = p;
00206     break;
00207   case 2:
00208     r = p; g = v; b = t;
00209     break;
00210    case 3:
00211      r = p; g = q; b = v;
00212      break;
00213   case 4:
00214     r = t;  g = p; b = v;
00215     break;
00216   default:    // case 5:
00217     r = v; g = p; b = q;
00218     break;
00219   }
