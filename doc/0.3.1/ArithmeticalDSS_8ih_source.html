00915 typename DGtal::ArithmeticalDSS<TIterator,TInteger,connectivity>::PointD
00916 DGtal::ArithmeticalDSS<TIterator,TInteger,connectivity>::project( const Point & m ) const
00917 {
00918   //casts
00919   double a = (double) NumberTraits<TInteger>::castToInt64_t(myA);
00920   double b = (double) NumberTraits<TInteger>::castToInt64_t(myB);
00921   double mu = (double) NumberTraits<TInteger>::castToInt64_t(myMu);
00922   double omega = (double) NumberTraits<TInteger>::castToInt64_t(myOmega);
00923   double xm = (double) NumberTraits<TInteger>::castToInt64_t(m[0]);
00924   double ym = (double) NumberTraits<TInteger>::castToInt64_t(m[1]);
00925 
00926   double alpha = ( mu + (omega - 1.0)/2.0 );
00927   double d2 = ( a * a + b * b );
00928   double s = b * xm + a * ym;
00929   double xp = ( b * s + a * alpha ) / d2;
00930   double yp = ( a * s - b * alpha ) / d2;
00931   return PointD( xp, yp );
00932 }
