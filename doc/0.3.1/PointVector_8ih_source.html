00612 inline
00613 double
00614 DGtal::PointVector<dim, TComponent>::norm (const
00615             typename Self::NormType aType ) const
00616 {
00617   double tmp = 0.0;
00618 
00619   ASSERT ( dim > 0 );
00620 
00621   switch ( aType )
00622     {
00623     case L_2:
00624       for ( Dimension i = 0; i < dimension; i++ )
00625         tmp += NumberTraits<Component>::castToDouble(at ( i )) *
00626     NumberTraits<Component>::castToDouble(at ( i ));
00627       tmp = ( double ) sqrt ( tmp );
00628       break;
00629     case L_1:
00630       for ( Dimension i = 0; i < dimension; i++ )
00631         tmp += fabs (  NumberTraits<Component>::castToDouble(at( i ) ));
00632       break;
00633     case L_infty:
00634       tmp = NumberTraits<Component>::castToDouble( at( 0 ));
00635       for ( Dimension i = 1; i < dimension; i++ )
00636         if ( tmp < fabs ( NumberTraits<Component>::castToDouble(at ( i ) ) ))
00637           tmp = fabs ( NumberTraits<Component>::castToDouble(at ( i ) ));
00638       break;
00639   }
00640   return tmp;
00641 }
