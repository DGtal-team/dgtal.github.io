00867 template <typename TInteger>
00868 void DGtal::FreemanChain<TInteger>::innerContour( 
00869     FreemanChain & aInnerChain,
00870     std::vector<unsigned int> & aOuter2inner,
00871     std::vector<unsigned int> & aInner2outer,
00872     const FreemanChain & aOuterChain,
00873     bool ccw )
00874 {
00875   unsigned int nb = aOuterChain.chain.size();
00876   unsigned int j = 0;
00877   aOuter2inner.clear();
00878   aOuter2inner.reserve( nb );
00879   // aInnerChain.chain.reserve( nb + 4 );
00880   aInnerChain.chain = "";
00881   aInner2outer.clear();
00882   aInner2outer.reserve( nb + ( ccw ? 4 : -4 ) );
00883   int dx0=0, dy0=0;
00884   int dx1=0, dy1=0;
00885   FreemanChain<TInteger>::displacement( dx0, dy0, aOuterChain.code( 0 ) );
00886   int turn = ccw ? 1 : 3;
00887   //FreemanChain<TInteger>::displacement( dx1, dy1, ( aOuterChain.code( 0 ) + turn ) % 4 );
00888   FreemanChain<TInteger>::displacement( dx1, dy1, addToCode( aOuterChain.code( 0 ) , turn ) );
00889   dx0 += dx1;
00890   dy0 += dy1;
00891   aInnerChain.x0 = dx0 > 0 ? aOuterChain.x0 : aOuterChain.x0 - 1;
00892   aInnerChain.y0 = dy0 > 0 ? aOuterChain.y0 : aOuterChain.y0 - 1;
00893 
00894   ConstIterator it_begin = aOuterChain.begin();
00895   ConstIterator it = it_begin;
00896   it.next();
00897   for ( unsigned int i = 0; i < nb; ++i )
00898   {
00899     // Check if contour is open.
00900     // cerr << "i=" << i << " code=" << aOuterChain.code( i ) << endl;
00901     switch ( movement( aOuterChain.code( i ), aOuterChain.code( ( i + 1 ) % nb ), ccw ) )
00902     {
00903       case '0':
00904         // contour going in then out.
00905         aInnerChain.chain += aOuterChain.chain[ i ];
00906         //aInnerChain.chain += ( ( ( (unsigned int) ( aOuterChain.chain[ i ] - '0' )
00907         //        + ( ccw ? 3 : 1 ) ) )
00908         //    % 4 ) + '0';
00909         aInnerChain.chain += addToCode ( aOuterChain.chain[ i ],  (ccw) ? 3 : 1  );
00910         aInnerChain.chain += aOuterChain.chain[ ( i + 1 ) % nb ];
00911         aOuter2inner.push_back( j );
00912         aInner2outer.push_back( i );
00913         aInner2outer.push_back( i + 1 );
00914         aInner2outer.push_back( i + 1 );
00915         j += 3;
00916         break;
00917 
00918       case '1':
00919         // contour turning toward its inside.
00920         aOuter2inner.push_back( j );
00921         break;
00922 
00923       case '2':
00924         // contour going straight ahead
00925         aInnerChain.chain += aOuterChain.chain[ i ];
00926         aOuter2inner.push_back( j );
00927         aInner2outer.push_back( i );
00928         ++j;
00929         break;
00930 
00931       case '3':
00932         // contour turning toward its outside.
00933         aInnerChain.chain += aOuterChain.chain[ i ];
00934         aInnerChain.chain += aOuterChain.chain[ ( i + 1 ) % nb ];
00935         aOuter2inner.push_back( j );
00936         aInner2outer.push_back( i );
00937         aInner2outer.push_back( i + 1 );
00938         j += 2;
00939         break;
00940     }
00941 
00942     // Advances along contour and check if it is a closed contour.
00943     it.next();
00944     if ( ( i == nb - 1 )
00945         && ( *it_begin != *it ) )
00946       // freeman chain is *not* a closed loop.
00947     {
00948       aInnerChain.chain += aOuterChain.chain[ i ];
00949       aOuter2inner.push_back( j );
00950       aInner2outer.push_back( i );
00951       ++i;
00952       ++j;
00953       break;
00954     }
00955   }
00956 }
