00513 template <typename TDigitalTopology, typename TDigitalSet>
00514 template <typename OutputObjectIterator>
00515 inline
00516 typename DGtal::Object<TDigitalTopology, TDigitalSet>::Size
00517 DGtal::Object<TDigitalTopology, TDigitalSet>
00518 ::writeComponents( OutputObjectIterator & it ) const
00519 {
00520   Size nb_components = 0;
00521   if ( pointSet().empty() )
00522   {
00523     myConnectedness = CONNECTED;
00524     return nb_components;
00525   }
00526   else
00527     if ( connectedness() == CONNECTED )
00528     {
00529       *it++ = *this;
00530       return 1;
00531     }
00532   typedef typename DigitalSet::ConstIterator DigitalSetConstIterator;
00533   DigitalSetConstIterator it_object = pointSet().begin();
00534   Point p( *it_object++ );
00535   Expander<Object> expander( *this, p );
00536   while ( expander.nextLayer() )
00537     ;
00538   Object component( myTopo, expander.core(), CONNECTED );
00539   *it++ = component;
00540   ++nb_components;
00541   DigitalSet visited( expander.core() );
00542   while ( it_object != pointSet().end() )
00543   {
00544     p = *it_object++;
00545     if ( visited.find( p ) == visited.end() )
00546     {
00547       Expander<Object> expander2( *this, p );
00548       while ( expander2.nextLayer() )
00549         ;
00550       Object component2( myTopo, expander2.core(), CONNECTED );
00551       *it++ = component2;
00552       ++nb_components;
00553       visited += expander2.core();
00554     }
00555   }
00556   myConnectedness = nb_components == 1 ? CONNECTED : DISCONNECTED;
00557   return nb_components;
00558 }

00618 DGtal::Object<TDigitalTopology, TDigitalSet>
00619 ::geodesicNeighborhood
00620 ( const TAdjacency & adj, const Point & p, unsigned int k ) const
00621 {
00622   // Local types.
00623   typedef std::vector<Point> Container;
00624   typedef MetricAdjacency<Space, Space::dimension> AlphaAdjacency;
00625   typedef MetricAdjacency<Space, 1> OmegaAdjacency;
00626   typedef DGtal::DigitalTopology<TAdjacency, OmegaAdjacency> LocalTopology;
00627   typedef typename Container::const_iterator ContainerConstIterator;
00628   typedef typename DigitalSet::ConstIterator DigitalSetConstIterator;
00629   typedef Object<LocalTopology, SmallSet> LocalObject;
00630   typedef HyperRectDomain<Space> LocalDomain;
00631 
00632   DigitalSetDomain<DigitalSet> limitedX( *myPointSet );
00633   AlphaAdjacency alpha;
00634   OmegaAdjacency omega;
00635   // Intermediate container that is fast writable.
00636   Container local_points;
00637   std::back_insert_iterator< Container > back_ins_it( local_points );
00638   alpha.writeProperNeighborhood( p, back_ins_it, limitedX.predicate() );
00639 
00640   // Construct local domain.
00641   Point p1( p );
00642   for ( typename Point::Iterator it = p1.begin(); it != p1.end(); ++it )
00643     --(*it);
00644   Point p2( p );
00645   for ( typename Point::Iterator it = p2.begin(); it != p2.end(); ++it )
00646     ++(*it);
00647   LocalDomain aDomain( p1, p2 );
00648 
00649   // Construct local X.
00650   LocalTopology aTopology( adj, omega );
00651   LocalObject X( aTopology, aDomain );
00652   X.pointSet().insertNew( local_points.begin(), local_points.end() );
00653 
00654   // A neighborhood is small, so is defined the digital object.
00655   typename LocalObject::SmallObject neighAdj = X.properNeighborhood( p );
00656   Expander<LocalObject> expander( X,
00657       neighAdj.pointSet().begin(),
00658       neighAdj.pointSet().end() );
00659   for ( unsigned int i = 1; ( i < k ) && ( ! expander.finished() ); ++i )
00660     expander.nextLayer();
00661 
00662   SmallObject geodesicN( this->topology(), expander.core() );
