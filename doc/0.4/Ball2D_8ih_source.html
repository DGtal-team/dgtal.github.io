00084 {

00087   
00088   double angle = 0.0;
00089   if ( ( p.at( 0 ) == 0.0 ) && ( p.at( 1 ) == 0.0 ) )
00090     return angle;
00091   if ( p.at( 0 ) >= p.at( 1 ) )
00092     {
00093       if ( p.at( 0 ) >= -p.at( 1 ) )
00094   angle = atan( p.at( 1 ) / p.at( 0 ) );
00095       else
00096   angle = 1.5* M_PI + atan( - p.at( 0 ) / p.at( 1 ) );
00097     }
00098   else // ( p.at( 0 ) >= p.at( 1 ) )
00099     {
00100       if ( p.at( 0 ) >= -p.at( 1 ) )
00101   angle = 0.5*M_PI - atan( p.at( 0 ) / p.at( 1 ) );
00102       else
00103   angle = M_PI + atan( p.at( 1 ) / p.at( 0 ) );
00104     }
00105   angle = ( angle < 0.0 ) ? angle + 2*M_PI : angle;
