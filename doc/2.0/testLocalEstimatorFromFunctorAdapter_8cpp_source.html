74struct ImplicitDigitalEllipse3 {
75 typedef TPoint3 Point;

78 : myA( a ), myB( b ), myC( c )
79 {}
80 inline
81 bool operator()( const TPoint3 & p ) const
82 {
83 double x = ( (double) p[ 0 ] / myA );
84 double y = ( (double) p[ 1 ] / myB );
