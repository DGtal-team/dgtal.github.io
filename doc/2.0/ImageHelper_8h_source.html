373 ImageToConstantFunctor( ConstAlias< Image > anImage,
374 ConstAlias< PointPredicate > aPointPred,
375 Value aVal = NumberTraits< Value >::ONE,

391 if ((myImage->domain().isInside(aPoint)))

403 }
404 else
405 {
406 if( myPointPred->operator()(aPoint) )
407 {
408 return myVal;
409 }
410 else
411 {
412 return NumberTraits<Value>::ZERO;
413 }
414 }
415 }
416 else
417 {
418 return NumberTraits<Value>::ZERO;
419 }
420 }
